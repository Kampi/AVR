522 void USART_SetBaudrate(USART_t* Device, const uint32_t Baudrate, const uint32_t Clock, const int8_t BSCALE, const Bool_t DoubleSpeed)
523 {
524  uint16_t BSEL = 0x00;
525 
526  if(Baudrate < (Clock / ((DoubleSpeed + 0x01) << 0x03)))
527  {
528  BSEL = (Clock / (Baudrate << 0x01)) - 1;
529  }
530 
531  if(BSCALE >= 0x00)
532  {
533  BSEL = (uint16_t)(((float)Clock / ((float)(0x01 << BSCALE) * Baudrate * (0x10 >> DoubleSpeed))) - 0x01);
534  }
535  else
536  {
537  BSEL = (uint16_t)((0x01 << (~BSCALE + 1)) * (((float)Clock / ((float)(0x10 >> DoubleSpeed) * Baudrate)) - 0x01));
538  }
539 
540  Device->CTRLB = (Device->CTRLB & ~((0x01 << 0x02))) | (DoubleSpeed << 0x02);
541  Device->BAUDCTRLB = ((BSCALE & 0x0F) << 0x04) | ((BSEL & 0xF00) >> 0x08);
542  Device->BAUDCTRLA = BSEL & 0xFF;
543 }
