49 static inline void Timer0_InterruptHandler(const uint8_t Device, const Timer_CallbackType_t Callback)
50 {
51  DisableGlobalInterrupts();
52 
53  if(Callback & TIMER_OVERFLOW_INTERRUPT)
54  {
55  TCNT0 = Period;
56 
57  if(Timer0_Callbacks[Device].OverflowCallback != NULL)
58  {
59  Timer0_Callbacks[Device].OverflowCallback();
60  }
61  }
62  else if(Callback & TIMER_COMPMATCH_INTERRUPT)
63  {
64  if(Timer0_Callbacks[Device].CompareCallback != NULL)
65  {
66  Timer0_Callbacks[Device].CompareCallback();
67  }
68  }
69 
70  EnableGlobalInterrupts();
71 }

94 void Timer0_InstallCallback(Timer0_InterruptConfig_t* Config)
95 {
96  if(Config->Source & TIMER_COMPMATCH_INTERRUPT)
97  {
98  Timer0_Callbacks[0].CompareCallback = Config->Callback;
99 
100  TIMSK |= (0x01 << OCIE0);

102 
103  if(Config->Source & TIMER_OVERFLOW_INTERRUPT)

105  Timer0_Callbacks[0].OverflowCallback = Config->Callback;
