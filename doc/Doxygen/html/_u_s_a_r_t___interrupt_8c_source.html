64 static void __USART_InterruptHandler(const uint8_t Device, const uint8_t Channel, const USART_CallbackType_t Callback)
65 {
66  if(__USART_IsSPI[Device][Channel] == TRUE)
67  {
68  if(Callback == USART_TXC_INTERRUPT)
69  {
70  if(__SPI_Messages[Device][Channel].BytesProcessed < (__SPI_Messages[Device][Channel].Length - 1))
71  {
72  ((USART_t*)(__SPI_Messages[Device][Channel].Device))->DATA = __SPI_Messages[Device][Channel].BufferOut[++__SPI_Messages[Device][Channel].BytesProcessed];
73  }
74  else
75  {
76  USART_SPI_DeselectDevice(__SPI_Messages[Device][Channel].Port, __SPI_Messages[Device][Channel].Pin);
77 
78  if(__USART_SPI_Callbacks[Device][Channel].CompleteInterrupt != NULL)
79  {
80  __USART_SPI_Callbacks[Device][Channel].CompleteInterrupt();
81  }
82 
83  __SPI_Messages[Device][Channel].Status = SPI_MESSAGE_COMPLETE;
84  }
85  }
86  else if(Callback == USART_RXC_INTERRUPT)
87  {
88  __SPI_Messages[Device][Channel].BufferIn[__SPI_Messages[Device][Channel].BytesProcessed] = ((USART_t*)(__SPI_Messages[Device][Channel].Device))->DATA;
89  }
90  }
91  else
92  {
93  if(Callback == USART_DRE_INTERRUPT)
94  {
95  if (__USART_Callbacks[Device][Channel].EmptyCallback)
96  {
97  __USART_Callbacks[Device][Channel].EmptyCallback();
98  }
99 
100  if(!(RingBuffer_IsEmpty(__USART_Messages[Device][Channel].Ptr_TxRingBuffer)))
101  {
102  __USART_Messages[Device][Channel].Device->DATA = RingBuffer_Load(__USART_Messages[Device][Channel].Ptr_TxRingBuffer);
103  }
104  else
105  {
106  __USART_Messages[Device][Channel].Device->CTRLA &= ~0x03;
107  }
108  }
109  else if(Callback == USART_TXC_INTERRUPT)
110  {
111  if (__USART_Callbacks[Device][Channel].TxCallback)
112  {
113  __USART_Callbacks[Device][Channel].TxCallback();
114  }
115  }
116  else if(Callback == USART_RXC_INTERRUPT)
117  {
118  uint8_t ReceivedByte = __USART_Messages[Device][Channel].Device->DATA;
119 
120  // Echo message if enabled
121  if(__USART_Echo[Device][Channel] == TRUE)
122  {
123  __USART_Messages[Device][Channel].Device->DATA = ReceivedByte;
124  }
125 
126  // Save the data in the buffer
127  RingBuffer_Save(&__USART_RxRingBuffer[Device][Channel], ReceivedByte);
128 
129  if (__USART_Callbacks[Device][Channel].RxCallback)
130  {
131  __USART_Callbacks[Device][Channel].RxCallback();
132  }
133  }
134  }
135 }
