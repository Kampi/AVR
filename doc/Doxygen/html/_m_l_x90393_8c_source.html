450 {
451  uint8_t Data[3];
452  uint8_t Resolution_Temp = Resolution & 0x03;
453  I2C_Error_t ErrorCode = I2C_NO_ERROR;
454 
455  ErrorCode = MLX90393_ReadRegister(MLX90393_REGISTER_2, Data);

458  return ErrorCode;

460

464 
465  uint16_t Config = (Data[1] << 0x08) | Data[2];
466 
467  // Set new resolution
468  if(Axis & MLX90393_AXIS_X)
469  {
470  Config |= Resolution_Temp << 0x09;
471  }
472 
473  if(Axis & MLX90393_AXIS_Y)
474  {
475  Config |= Resolution_Temp << 0x07;
476  }
477 
478  if(Axis & MLX90393_AXIS_Z)
479  {
480  Config |= Resolution_Temp << 0x05;
481  }
482 
483  return MLX90393_WriteRegister(MLX90393_REGISTER_2, Config, NULL);
