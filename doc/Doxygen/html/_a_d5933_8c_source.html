96 static float AD5933_CalcPhase(const float Real, const float Imag)
97 {
98  double Angle = 0.0;
99 
100  if(Imag >= 0)
101  {
102  Angle = atan2(Imag, Real);
103  }
104  else
105  {
106  Angle = 2 * M_PI + atan2(Imag, Real);
107  }
108 
109  if(UseDegree == TRUE)
110  {
111  Angle = (180.0 / M_PI) * Angle;
112  }
113 
114  return Angle;
115 }

367 const I2C_Error_t AD5933_GetData(ComplexNumber_t* Data)
368 {
369  uint8_t Buffer[4];
370  uint8_t Status = 0x00;
371  I2C_Error_t ErrorCode = I2C_NO_ERROR;
372 
373  while(!(Status & AD5933_STATUS_VALID_DATA))
374  {
375  ErrorCode = AD5933_ReadStatus(&Status);
376  if(ErrorCode != I2C_NO_ERROR)
377  {
378  return ErrorCode;
379  }
380  }
381 
382  Buffer[0] = AD5933_CMD_BLOCK_READ;
383  Buffer[1] = 4;
384  ErrorCode = I2CM_WRITEBYTE(AD5933_INTERFACE, AD5933_ADDRESS, AD5933_REGISTER_REAL, TRUE);
385  ErrorCode |= I2CM_WRITEBYTES(AD5933_INTERFACE, AD5933_ADDRESS, 2, Buffer, FALSE);
386  ErrorCode |= I2CM_READBYTES(AD5933_INTERFACE, AD5933_ADDRESS, sizeof(Buffer), Buffer, TRUE);
387  if(ErrorCode != I2C_NO_ERROR)
388  {
389  return ErrorCode;
390  }
391 
392  Data->Real = (Buffer[0] << 0x08) + Buffer[1];
393  Data->Imag = (Buffer[2] << 0x08) + Buffer[3];
394 
395  return ErrorCode;
396 }

580 {

583  uint8_t Status = 0x00;
584  I2C_Error_t ErrorCode = I2C_NO_ERROR;
585

590  }

619  {
620  return ErrorCode;
621  }
622  }
