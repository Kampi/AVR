47 static inline void I2C_InterruptHandler(const uint8_t Device)
48 {
49  DisableGlobalInterrupts();
50 
51  // Clear interrupt flag
52  TWCR |= 0x01 << TWINT;
53 
54  if(I2C_Callbacks[Device].TransferCompleteCallback != NULL)
55  {
56  I2C_Callbacks[Device].TransferCompleteCallback();
57  }
58 
59  EnableGlobalInterrupts();
60 }

74 void I2CM_SetBaudrate(const uint32_t Bitrate, const uint8_t Prescaler, const uint32_t Clock)
75 {
76  uint8_t Exp = 0x01;
77 
78  for(uint8_t i = 0x00; i < Prescaler; i++)
79  {
80  Exp = Exp << 0x02;
81  }
82 
83  TWBR = ((Clock / Bitrate) - 0x10) / (Exp << 0x01);
84 }

130 void I2C_SendStop(void)
131 {
132  TWCR = (0x01 << TWINT) | (0x01 << TWEN) | (0x01 << TWSTO);
133 
134  // Wait a bit to reduce errors when a Start-Condition follows a Stop-Condition
135  for(uint8_t Delay = 0x00; Delay < 0xFF; Delay++);
136 }

138 I2C_Status_t I2C_SendStart(void)
139 {
140  TWCR = (0x01 << TWINT) | (0x01 << TWEN) | (0x01 << TWSTA);
141 
142  while(!(TWCR & (0x01 << TWINT)));
143 
144  return I2C_ReadStatus();

165  TWCR = (0x01 << TWINT) | (0x01 << TWEN) | (0x01 << TWEA);

169

198  return I2C_START_ERROR;
