124  // Internal 32 kHz RC oscillator

127  break;
128  }

130  case(RTC_CLOCK_RCOSC32):

133  while(!(OSC.STATUS & OSC_RC32KRDY_bm));
134  break;
135  }
136  // External 32 kHz oscillator
137  case(RTC_CLOCK_TOSC):
138  case(RTC_CLOCK_TOSC32):
139  {

143  break;
144  }
145  // External clock

147  {
148  break;
149  }

152  break;
