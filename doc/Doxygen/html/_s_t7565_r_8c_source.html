99 static void ST7565R_WriteCommandBytes(const uint8_t* Data, const uint8_t Length)
100 {
101  SPIM_CHIP_SELECT(ST7565R_SS_PORT, ST7565R_SS_PIN);
102  GPIO_Clear(ST7565R_REGISTER_SELECT_PORT, ST7565R_REGISTER_SELECT_PIN);
103 
104  for(uint8_t i = 0x00; i < Length; i++)
105  {
106  SPIM_TRANSMIT(DISPLAY_INTERFACE, *Data++);

113 void Display_Init(SPIM_Config_t* Config)
114 {
115  const uint8_t Init[] = {
116  ST7565R_CMD_ADC_NORMAL,
117  ST7565R_CMD_DISPLAY_NORMAL,
118  ST7565R_CMD_REVERSE_SCAN_DIRECTION,
119  ST7565R_CMD_LCD_BIAS_1_DIV_6_DUTY33,
120  ST7565R_CMD_POWER_CTRL_ALL_ON,
121  ST7565R_CMD_BOOSTER_RATIO,
122  ST7565R_CMD_BOOSTER_RATIO_2X_3X_4X,
123  ST7565R_CMD_VOLTAGE_RESISTOR_RATIO_1,
124  ST7565R_CMD_ELECTRONIC_VOLUME_MODE_SET,
125  ST7565R_CMD_ELECTRONIC_VOLUME(DISPLAY_CONTRAST_MIN),
126  ST7565R_CMD_DISPLAY_ON,
127  };
128 
129  // Set display back light as output in low state
130  GPIO_SetDirection(ST7565R_BACKLIGHT_PORT, ST7565R_BACKLIGHT_PIN, GPIO_DIRECTION_OUT);
131  GPIO_Clear(ST7565R_BACKLIGHT_PORT, ST7565R_BACKLIGHT_PIN);
132 
133  // Set display reset as output in high state
134  GPIO_SetDirection(ST7565R_RESET_PORT, ST7565R_RESET_PIN, GPIO_DIRECTION_OUT);
135  GPIO_Set(ST7565R_RESET_PORT, ST7565R_RESET_PIN);
136 
137  // Set register select as output in high state
138  GPIO_SetDirection(ST7565R_REGISTER_SELECT_PORT, ST7565R_REGISTER_SELECT_PIN, GPIO_DIRECTION_OUT);
139  GPIO_Set(ST7565R_REGISTER_SELECT_PORT, ST7565R_REGISTER_SELECT_PIN);
140 
141  // Set display SS as output in high state
142  GPIO_SetDirection(ST7565R_SS_PORT, ST7565R_SS_PIN, GPIO_DIRECTION_OUT);
143  GPIO_Set(ST7565R_SS_PORT, ST7565R_SS_PIN);
144 
145  // Reset the display
146  ST7565R_HardwareReset();
147 
148  // Initialize SPI-Interface
149  if(Config != NULL)
150  {
151  SPIM_INIT(Config);
152  }
153 
154  // Set A0 to low state for "Command"
155  GPIO_Clear(ST7565R_REGISTER_SELECT_PORT, ST7565R_REGISTER_SELECT_PIN);
156 
157  ST7565R_WriteCommandBytes(Init, sizeof(Init));
158 
159  // Clear Display
160  ST7565R_Clear();
161 }
