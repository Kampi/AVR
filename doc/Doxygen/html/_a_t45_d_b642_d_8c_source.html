264 void AT45DB642D_Flush(AT45DB642D_Buffer_t Buffer, uint16_t PageAddress)
265 {
266  uint8_t BufferCommand = AT45DB642D_CMD_BUFFER1_TO_MEMORY_ERASE;
267 
268  // Wait until device is ready
269  while(!(AT45DB642D_ReadStatus() & 0x80));
270 
271  // Mask 10 address bits
272  uint16_t PageAddress_Temp = (PageAddress & 0x3FF);
273 
274  if(Buffer == AT45DB642D_BUFFER_2)
275  {
276  BufferCommand = AT45DB642D_CMD_BUFFER2_TO_MEMORY_ERASE;
277  }
278 
279  //AT45DB642D_SPI_CHIP_SELECT(FLASH_SS_PORT, FLASH_SS_PIN);
280  AT45DB642D_SPI_TRANSMIT(BufferCommand);
281  AT45DB642D_SPI_TRANSMIT(0x00);
282  AT45DB642D_SPI_TRANSMIT((PageAddress_Temp & 0xFF00) >> 0x08);
283  AT45DB642D_SPI_TRANSMIT((PageAddress_Temp & 0x00FF));
