51 static void __SPI_InterruptHandler(const uint8_t Device)
52 {
53  SPI_t* Device_Ptr = __SPIM_Messages[Device].Device;
54 
55  // Check if master or slave mode
56  if(__SPI_DeviceMode[Device] == SPI_MASTER)
57  {
58  // Check for error
59  if(Device_Ptr->STATUS & SPI_WRCOL_bm)
60  {
61  __SPIM_Messages[Device].Status = SPI_MESSAGE_ERROR;
62 
63  if(__SPI_Callbacks[Device].ErrorCallback != NULL)
64  {
65  __SPI_Callbacks[Device].ErrorCallback();
66  }
67  }
68  else
69  {
70  __SPIM_Messages[Device].BufferIn[__SPIM_Messages[Device].BytesProcessed] = Device_Ptr->DATA;
71 
72  if(__SPIM_Messages[Device].BytesProcessed < (__SPIM_Messages[Device].Length - 1))
73  {
74  Device_Ptr->DATA = __SPIM_Messages[Device].BufferOut[++__SPIM_Messages[Device].BytesProcessed];
75  }
76  else
77  {
78  __SPIM_Messages[Device].Status = SPI_MESSAGE_COMPLETE;
79 
80  if(__SPI_Callbacks[Device].CompleteInterrupt != NULL)
81  {
82  __SPI_Callbacks[Device].CompleteInterrupt();
83  }
84 
85  SPIM_DeselectDevice(__SPIM_Messages[Device].Port, __SPIM_Messages[Device].Pin);
86  }
87  }
88  }
89  else
90  {
91  __SPI__SlaveBuffer[Device].RxBuffer[__SPI__SlaveBuffer[Device].BytesProcessed] = SPIC.DATA;
92 
93  if(__SPI__SlaveBuffer[Device].BytesProcessed < (SPI_BUFFER_SIZE - 1))
94  {
95  SPIC.DATA = __SPI__SlaveBuffer[Device].TxBuffer[++__SPI__SlaveBuffer[Device].BytesProcessed];
96  }
97  else
98  {
99  __SPI__SlaveBuffer[Device].BytesProcessed = 0x00;
100  __SPI__SlaveBuffer[Device].Status = SPI_BUFFER_OVERFLOW;
101  }
102  }
103 }
