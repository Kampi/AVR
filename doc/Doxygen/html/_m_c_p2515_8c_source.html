257 static inline void MCP2515_Interrupthandler(void)
258 {
259  static CAN_Message_t Message;
260 
261  uint8_t Status = MCP2515_ReadRegister(MCP2515_REGISTER_CANINTF);
262 
263  if((Status & (0x01 << MCP2515_IF_RX0)) || (Status & (0x01 << MCP2515_IF_RX1)))
264  {
265  MCP2515_ReadMessage(&Message);
266 
267  // General message callback for all messages
268  if(MCP2515_Callbacks.MessageCallback != NULL)
269  {
270  MCP2515_Callbacks.MessageCallback(Message);
271  }
272 
273  // Message specific callbacks
274  if(Message.MessageType == MCP2515_STANDARD_DATA)
275  {
276  if(MCP2515_Callbacks.StandardDataCallback != NULL)
277  {
278  MCP2515_Callbacks.StandardDataCallback(Message);
279  }
280  }
281  else if(Message.MessageType == MCP2515_EXTENDED_DATA)
282  {
283  if(MCP2515_Callbacks.ExtendetDataCallback != NULL)
284  {
285  MCP2515_Callbacks.ExtendetDataCallback(Message);
286  }
287  }
288  else if(Message.MessageType == MCP2515_STANDARD_REMOTE)
289  {
290  if(MCP2515_Callbacks.StandardRemoteCallback != NULL)
291  {
292  MCP2515_Callbacks.StandardRemoteCallback(Message);
293  }
294  }
295  else if(Message.MessageType == MCP2515_EXTENDED_REMOTE)
296  {
297  if(MCP2515_Callbacks.ExtendedRemoteCallback != NULL)
298  {
299  MCP2515_Callbacks.ExtendedRemoteCallback(Message);
300  }
301  }
302  }

306 {

323  MCP2515_WriteRegister(MCP2515_REGISTER_CANCTRL, 0x08);
324

543 void MCP2515_ConfigFilter(const MCP2515_FilterConfig_t* Config)
544 {
545  // Get the current device mode
546  MCP2515_DeviceMode_t Mode = MCP2515_GetDeviceMode();
547 
548  // Change mode to configuration mode to change the filter setup
549  MCP2515_SetDeviceMode(MCP2515_CONFIG_MODE);
550 
551  uint8_t Identifier_Low = (Config->FilterID & 0x07) << 0x05;
552  uint8_t Identifier_High = Config->FilterID >> 0x03;
553 
554  uint8_t Mask_Low = (Config->FilterMask & 0x07) << 0x05;
555  uint8_t Mask_High = Config->FilterMask >> 0x03;
556 
557  MCP2515_WriteRegister(Config->Filter, Identifier_High);
558  MCP2515_WriteRegister(Config->Filter + 0x01, Identifier_Low);
559 
560  if((Config->Filter == MCP2515_RXF_0) | (Config->Filter == MCP2515_RXF_1))
561  {
562  MCP2515_WriteRegister(MCP2515_REGISTER_RXM0SIDH, Mask_High);
563  MCP2515_WriteRegister(MCP2515_REGISTER_RXM0SIDL, Mask_Low);
564  }
565  else if((Config->Filter == MCP2515_RXF_2) | (Config->Filter == MCP2515_RXF_3) | (Config->Filter == MCP2515_RXF_4) | (Config->Filter == MCP2515_RXF_5))
566  {
567  MCP2515_WriteRegister(MCP2515_REGISTER_RXM1SIDH, Mask_High);
568  MCP2515_WriteRegister(MCP2515_REGISTER_RXM1SIDL, Mask_Low);
569  }
570 
571  // Restore device mode
572  MCP2515_SetDeviceMode(Mode);
573 }

599 void MCP2515_ReadMessage(CAN_Message_t* Message)
600 {
601  uint8_t ID_High = 0x00;
602  uint8_t ID_Low = 0x00;
603  uint8_t Length = 0x00;
604  uint8_t ReceiveBufferAddress = 0x00;
605 
606  // Get the receive status
607  uint8_t RxStatus = MCP2515_ReadRxStatus();
608 
609  // Check witch receive buffer contains the message
610  if(RxStatus & 0x40)
611  {
612  ReceiveBufferAddress = MCP2515_CMD_READRX0;
613  }
614  else if(RxStatus & 0x80)
615  {
616  ReceiveBufferAddress = MCP2515_CMD_READRX1;
617  }
618 
619  MCP2515_SPI_CHIP_SELECT();
620 
621  MCP2515_SPI_TRANSMIT(ReceiveBufferAddress);
622 
623  // Get the message ID
624  ID_High = (MCP2515_SPI_TRANSMIT(0xFF)) << 0x03;
625  ID_Low = (MCP2515_SPI_TRANSMIT(0xFF)) >> 0x05;
626  Message->MessageID = ID_High | ID_Low;
627 
628  // Read the two empty bytes
629  MCP2515_SPI_TRANSMIT(0xFF);
630  MCP2515_SPI_TRANSMIT(0xFF);
631 
632  // Get the message length
633  Length = MCP2515_SPI_TRANSMIT(0xFF);
634 
635  // Check message type
636  uint8_t MessageType = (RxStatus & 0x18) >> 0x03;
637 
638  switch(MessageType)
639  {
640  case(MCP2515_STANDARD_DATA):
641  {
642  Message->MessageLength = Length;
643  Message->MessageType = MessageType;
644 
645  for(uint8_t i = 0; i < Message->MessageLength; i++)
646  {
