50 void USART_SPI_Init(SPIM_Config_t* Config)
51 {
52  uint8_t ID = 0x00;
53  uint8_t Channel = 0x00;
54 
55  // Get the USART device for USART-SPI mode
56  USART_t* Device = (USART_t*)Config->Device;
57 
58  USART_PowerEnable(Config->Device);
59 
60  // Disable Rx
61  Device->CTRLB &= ~USART_RXEN_bm;
62 
63  if(Device == &USARTC0)
64  {
65  ID = USARTC_ID;
66  Channel = 0x00;
67  }
68  else if(Device == &USARTD0)
69  {
70  ID = USARTD_ID;
71  Channel = 0x00;
72  }
73  else if(Device == &USARTE0)
74  {
75  ID = USARTE_ID;
76  Channel = 0x00;
77  }
78  #if(MCU_NAME == MCU_NAME_ATXMEGA256A3BU)
79  else if(Device == &USARTC1)
80  {
81  ID = USARTC_ID;
82  Channel = 0x01;
83  }
84 
85  else if(Device == &USARTD1)
86  {
87  ID = USARTD_ID;
88  Channel = 0x01;
89  }
90  else if(Device == &USARTF0)
91  {
92  ID = USARTF_ID;
93  Channel = 0x00;
94  }
95  #endif
96 
97  __USART_IsSPI[ID][Channel] = TRUE;
98 
99  if(Config->Device == &USARTD0)
100  {
101  GPIO_SetDirection(&PORTD, USART_TX0_PIN, GPIO_DIRECTION_OUT);
102  GPIO_Set(&PORTD, USART_TX0_PIN);
103  GPIO_SetDirection(&PORTD, USART_XCK0_PIN, GPIO_DIRECTION_OUT);
104  GPIO_Set(&PORTD, USART_XCK0_PIN);
105  GPIO_SetDirection(&PORTD, USART_RX0_PIN, GPIO_DIRECTION_IN);
106 
107  // Setup INVEN
108  if((Config->Mode == SPI_MODE_2) || (Config->Mode == SPI_MODE_3))
109  {
110  GPIO_SwitchInvert(&PORTD, (0x01 << SPI_SCK_PIN), TRUE);
111  }
112  else
113  {
114  GPIO_SwitchInvert(&PORTD, (0x01 << SPI_SCK_PIN), FALSE);
115  }
116  }
117 
118  // Enable USART master SPI
119  USART_SetDeviceMode(Device, 0x03);
120 
121  // Setup Clock Phase
122  if((Config->Mode == SPI_MODE_1) || (Config->Mode == SPI_MODE_3))
123  {
124  Device->CTRLC |= 0x02;
125  }
126  else
127  {
128  Device->CTRLC &= ~0x02;
129  }
130 
131  // Setup data order
132  if(Config->DataOrder == SPI_DATAORDER_LSB_FIRST)
133  {
134  Device->CTRLC |= 0x04;
135  }
136  else
137  {
138  Device->CTRLC &= ~0x04;
139  }
140 
141  USART_SPI_SetClockRate(Device, Config->SPIClock, SysClock_GetClock(), Config->EnableDoubleSpeed);
142 
143  // Enable Rx and Tx
144  Device->CTRLB |= (USART_RXEN_bm | USART_TXEN_bm);
145 }

197 {
198  while(!(Device->STATUS & USART_DREIF_bm));
199  Device->DATA = Data;

228  }
229 
230  #if(MCU_NAME == MCU_NAME_ATXMEGA256A3BU)
231  else if(Device == &USARTF0)
232  {
233  ID = USARTF_ID;
234  Channel = 0x00;
235  }
236  else if(Device == &USARTD1)
237  {
238  ID = USARTD_ID;
239  Channel = 0x01;
240  }
241  else if(Device == &USARTC1)
242  {
243  ID = USARTC_ID;
244  Channel = 0x01;
245  }
246  #endif
247 
248  // Save the message object
249  __SPI_Messages[ID][Channel] = *Message;
250 
251  // Reset the read and write counter
252  __SPI_Messages[ID][Channel].BytesProcessed = 0x00;
253 
254  // Wait for active transmissions
255  if(__SPI_Messages[ID][Channel].Status != SPI_MESSAGE_COMPLETE)
256  {
257  return __SPI_Messages[ID][Channel].Status;
258  }
259 
260  __SPI_Messages[ID][Channel].Status = 0x00;
261 
262  // Select the device
263  USART_SPI_SelectDevice(__SPI_Messages[ID][Channel].Port, __SPI_Messages[ID][Channel].Pin);
264 
265  // Send the first byte
266  ((USART_t*)(__SPI_Messages[ID][Channel].Device))->DATA = __SPI_Messages[ID][Channel].BufferOut[__SPI_Messages[ID][Channel].BytesProcessed];
267 
268  return SPI_MESSAGE_PENDING;
269 }

426 void USART_SPI_SetClockRate(USART_t* Device, const uint32_t Baudrate, const uint32_t Clock, const Bool_t DoubleSpeed)
427 {
428  uint16_t BaudValue = 0x00;
429 
430  if(Baudrate < (Clock >> 0x01))
431  {
432  BaudValue = (Clock / (Baudrate << 0x01)) - 0x01;
433  }
434 
435  Device->CTRLB = (Device->CTRLB & ~((0x01 << 0x02))) | (DoubleSpeed << 0x02);
436  Device->BAUDCTRLB = (uint8_t)((~USART_BSCALE_gm) & (BaudValue >> 0x08));
437  Device->BAUDCTRLA = (uint8_t)(BaudValue);
438 }
