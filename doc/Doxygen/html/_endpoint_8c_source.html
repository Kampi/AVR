41 Bool_t Endpoint_Configure(const uint8_t Address, const Endpoint_Type_t Type, const Endpoint_Size_t Size, const uint8_t Banks)
42 {
43  uint8_t Address_Temp = Address & 0x0F;
44 
45  if((Address_Temp & 0x07) > USB_TOTAL_ENDPOINTS)
46  {
47  return FALSE;
48  }
49 
50  // Allocate the memory for the endpoints
51  for(uint8_t i = Address_Temp; i < USB_TOTAL_ENDPOINTS; i++)
52  {
53  uint8_t UECFG0X_Temp;
54  uint8_t UECFG1X_Temp;
55  uint8_t UEIENX_Temp;
56 
57  Endpoint_Select(i);
58 
59  if(i == Address_Temp)
60  {
61  // Configure UECFG0X-register
62  UECFG0X_Temp = (Type << EPTYPE0);
63 
64  if(Address & ENDPOINT_DIR_MASK_IN)
65  {
66  UECFG0X_Temp |= (0x01 << EPDIR);
67  }
68 
69  // Configure UECFG1X-register
70  UECFG1X_Temp = (0x01 << ALLOC);
71 
72  if(Banks > 0x01)
73  {
74  UECFG1X_Temp |= (0x01 << EPBK0);
75  }
76 
77  if((Size == 0x06) || (Size == 0x07))
78  {
79  UECFG1X_Temp &= ~((0x01 << EPSIZE2) | (0x01 << EPSIZE1) | (0x01 << EPSIZE0));
80  }
81  else
82  {
83  UECFG1X_Temp |= (Size << EPSIZE0);
84  }
85 
86  // Configure UEIENX-register
87  UEIENX_Temp = 0x00;
88  }
89  else
90  {
91  UECFG0X_Temp = UECFG0X;
92  UECFG1X_Temp = UECFG1X;
93  UEIENX_Temp = UEIENX;
94  }
95 
96  // Continue if the chosen endpoint is allocated already
97  if(!(UECFG1X_Temp & (0x01 << ALLOC)))
98  {
99  continue;
100  }
101 
102  // Disable the selected endpoint
103  Endpoint_Disable();
104 
105  // Clear the ALLOC-bit, so the endpoints will slide down
106  UECFG1X &= ~(0x01 << ALLOC);
107 
108  // Configure and activate the endpoint
109  // See figure 23-2 in the device data sheet
110  Endpoint_Enable();
111  UECFG0X = UECFG0X_Temp;
112  UECFG1X = UECFG1X_Temp;
113  UEIENX = UEIENX_Temp;
114 
115  if(!(UESTA0X & (0x01 << CFGOK)))
116  {
117  return FALSE;
118  }
119  }
120 
121  // Select the configured endpoint
122  Endpoint_Select(Address_Temp);
123 
124  return TRUE;
125 }
