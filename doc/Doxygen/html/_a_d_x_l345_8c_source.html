129 static void __ADXL345_InterruptHandler(const uint8_t Channel)
130 {
131  uint8_t Status = 0x00;
132 
133  if((ADXL345_READBYTE(ADXL345_REGISTER_INT_SOURCE, &Status) == I2C_NO_ERROR))
134  {
135  if(Status & ADXL345_DATARDY_INTERRUPT)
136  {
137  if(__ADXL345_Callbacks[Channel].DataRdyCallback != NULL)
138  {
139  __ADXL345_Callbacks[Channel].DataRdyCallback();
140  }
141  }
142 
143  if(Status & ADXL345_SINGLE_TAP_INTERRUPT)
144  {
145  if(__ADXL345_Callbacks[Channel].SingleTapCallback != NULL)
146  {
147  __ADXL345_Callbacks[Channel].SingleTapCallback();
148  }
149  }
150 
151  if(Status & ADXL345_DOUBLE_TAP_INTERRUPT)
152  {
153  if(__ADXL345_Callbacks[Channel].DoubleTapCallback != NULL)
154  {
155  __ADXL345_Callbacks[Channel].DoubleTapCallback();
156  }
157  }
158 
159  if(Status & ADXL345_ACTIVITY_INTERRUPT)
160  {
161  if(__ADXL345_Callbacks[Channel].ActivityCallback != NULL)
162  {
163  __ADXL345_Callbacks[Channel].ActivityCallback();
164  }
165  }
166 
167  if(Status & ADXL345_INACTIVITY_INTERRUPT)
168  {
169  if(__ADXL345_Callbacks[Channel].InactivityCallback != NULL)
170  {
171  __ADXL345_Callbacks[Channel].InactivityCallback();
172  }
173  }
174 
175  if(Status & ADXL345_FREEFALL_INTERRUPT)
176  {
177  if(__ADXL345_Callbacks[Channel].FreeFallCallback != NULL)
178  {
179  __ADXL345_Callbacks[Channel].FreeFallCallback();
180  }
181  }
182 
183  if(Status & ADXL345_WATERMARK_INTERRUPT)
184  {
185  //ADXL345_FIFOClear(__ADXL345_FIFO);
186 
187  if(__ADXL345_Callbacks[Channel].WatermarkCallback != NULL)

195  if(__ADXL345_Callbacks[Channel].OverrunCallback != NULL)

198  }

569 const const I2C_Error_t ADXL345_SetDeviceMode(const ADXL345_DeviceMode_t Mode)
570 {
571  uint8_t BW_Rate = 0x00;
572  uint8_t Power_Ctrl = 0x00;
573  uint8_t Data_Format = 0x00;
574  I2C_Error_t ErrorCode = I2C_NO_ERROR;
575 
576  switch(Mode)
577  {
578  case ADXL345_MODE_STANDBY:
579  {
580  // Clear the LP bit
581  ErrorCode = ADXL345_READBYTE(ADXL345_REGISTER_BW_RATE, &BW_Rate);
582  if(ErrorCode != I2C_NO_ERROR)
583  {
584  return ErrorCode;
585  }
586 
587  BW_Rate &= ~(0x01 << ADXL345_LOW_POWER);
588 
589  ErrorCode = ADXL345_WRITEBYTE(ADXL345_REGISTER_BW_RATE, &BW_Rate);
590  if(ErrorCode != I2C_NO_ERROR)
591  {
592  return ErrorCode;
593  }
594 
595  // Clear sleep, auto sleep and measure bit
596  ErrorCode = ADXL345_READBYTE(ADXL345_REGISTER_POWER_CTL, &Power_Ctrl);
597  if(ErrorCode != I2C_NO_ERROR)
598  {
599  return ErrorCode;
600  }
601 
602  Power_Ctrl &= ~((0x01 << ADXL345_AUTO_SLEEP) | (0x01 << ADXL345_MEASURE) | (0x01 << ADXL345_SLEEP));
603 
604  ErrorCode = ADXL345_WRITEBYTE(ADXL345_REGISTER_POWER_CTL, &Power_Ctrl);
605  if(ErrorCode != I2C_NO_ERROR)
606  {
607  return ErrorCode;
608  }
609 
610  // Clear the self test bit
611  ErrorCode = ADXL345_READBYTE(ADXL345_REGISTER_DATA_FORMAT, &Data_Format);
612  if(ErrorCode != I2C_NO_ERROR)
613  {
614  return ErrorCode;
615  }
616 
617  Data_Format &= ~(0x01 << ADXL345_SELFTEST);
618 
619  ErrorCode = ADXL345_WRITEBYTE(ADXL345_REGISTER_DATA_FORMAT, &Data_Format);
620  if(ErrorCode != I2C_NO_ERROR)
621  {
622  return ErrorCode;
623  }
624 
625  break;
626  }
627  case ADXL345_MODE_LP:
628  {
629  // Set the LP bit
630  ErrorCode = ADXL345_READBYTE(ADXL345_REGISTER_BW_RATE, &BW_Rate);
631  if(ErrorCode != I2C_NO_ERROR)
632  {
633  return ErrorCode;
634  }
635 
636  BW_Rate |= (0x01 << ADXL345_LOW_POWER);
637 
638  ErrorCode = ADXL345_WRITEBYTE(ADXL345_REGISTER_BW_RATE, &BW_Rate);
639  if(ErrorCode != I2C_NO_ERROR)
640  {
641  return ErrorCode;
642  }
643 
644  break;
645  }
646  case ADXL345_MODE_SLEEP:
647  {
648  ErrorCode = ADXL345_READBYTE(ADXL345_REGISTER_POWER_CTL, &Power_Ctrl);
649  if(ErrorCode != I2C_NO_ERROR)
650  {
651  return ErrorCode;
652  }
653 
654  Power_Ctrl &= ~0x03;
655  Power_Ctrl |= ((0x01 << ADXL345_AUTO_SLEEP) | (0x01 << ADXL345_SLEEP));
656 
657  ErrorCode = ADXL345_WRITEBYTE(ADXL345_REGISTER_POWER_CTL, &Power_Ctrl);
658  if(ErrorCode != I2C_NO_ERROR)
659  {
660  return ErrorCode;
661  }
662 
663  break;
664  }
665  case ADXL345_MODE_MEASURE:
666  {
667  ErrorCode = ADXL345_READBYTE(ADXL345_REGISTER_POWER_CTL, &Power_Ctrl);
668  if(ErrorCode != I2C_NO_ERROR)
669  {
670  return ErrorCode;
671  }
672 
673  Power_Ctrl |= (0x01 << ADXL345_MEASURE);
674 
675  ErrorCode = ADXL345_WRITEBYTE(ADXL345_REGISTER_POWER_CTL, &Power_Ctrl);
676  if(ErrorCode != I2C_NO_ERROR)
677  {
678  return ErrorCode;
679  }
680 
681  for(uint16_t i = 0x00; i < 0x00; i++);
682 
683  break;
684  }
685 
686  case ADXL345_MODE_SELFTEST:
687  {
688  ErrorCode = ADXL345_READBYTE(ADXL345_REGISTER_DATA_FORMAT, &Data_Format);
689  if(ErrorCode != I2C_NO_ERROR)
690  {
691  return ErrorCode;
692  }
693 
694  Data_Format |= (0x01 << ADXL345_SELFTEST);
695 
696  ErrorCode = ADXL345_WRITEBYTE(ADXL345_REGISTER_DATA_FORMAT, &Data_Format);
697  if(ErrorCode != I2C_NO_ERROR)
698  {
699  return ErrorCode;
700  }
701 
702  break;
703  }
704  }
705 
706  return ErrorCode;
707 }
