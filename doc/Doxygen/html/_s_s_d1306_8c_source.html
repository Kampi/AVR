123 void Display_Init(SPIM_Config_t* Config)
124 {
125  // Set Data/Command pin as output in low state
126  GPIO_SetDirection(SSD1306_DATA_CMD_PORT, SSD1306_DATA_CMD_PIN, GPIO_DIRECTION_OUT);
127  GPIO_Set(SSD1306_DATA_CMD_PORT, SSD1306_DATA_CMD_PIN);
128 
129  // Set display reset as output in high state
130  GPIO_SetDirection(SSD1306_RESET_PORT, SSD1306_RESET_PIN, GPIO_DIRECTION_OUT);
131  GPIO_Set(SSD1306_RESET_PORT, SSD1306_RESET_PIN);
132 
133  // Set display SS as output in high state
134  GPIO_SetDirection(SSD1306_SS_PORT, SSD1306_SS_PIN, GPIO_DIRECTION_OUT);
135  GPIO_Set(SSD1306_SS_PORT, SSD1306_SS_PIN);
136 
137  Display_Reset();
138 
139  // Initialize SPI-Interface
140  if(Config != NULL)
141  {
142  SPIM_INIT(Config);
143  }
144 
145  SSD1306_WriteCommandBytes(DisplayInitialization, sizeof(DisplayInitialization));
146 }

162 void Display_SetColumn(const uint8_t Column)
163 {
164  // Use only the lower 4 bits, because the controller supports only 128 columns
165  uint8_t Temp = Column & 0x7F;
166  SSD1306_WriteCommand(SSD1306_CMD_HIGH_COL(Temp >> 0x04));
167  SSD1306_WriteCommand(SSD1306_CMD_LOW_COL(Temp & 0x0F));
168 }
