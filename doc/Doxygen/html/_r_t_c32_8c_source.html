83 void RTC32_ChangeInterruptLevel(const RTC32_CallbackType_t Callback, const Interrupt_Level_t InterruptLevel)
84 {
85  if(Callback & RTC32_OVFL_INTERRUPT)
86  {
87  RTC32.INTCTRL |= (RTC32.INTCTRL & (~InterruptLevel)) | InterruptLevel;
88  }
89 
90  if(Callback & RTC32_COMP_INTERRUPT)
91  {
92  RTC32.INTCTRL |= (RTC32.INTCTRL & (~(InterruptLevel << 0x02))) | (InterruptLevel << 0x02);
