75 static void __I2CM_WriteHandler(const uint8_t Device)
76 {
77  // Abort transmission of slave has send an NACK
78  if(__I2CM_Messages[Device].Device->MASTER.STATUS & TWI_MASTER_RXACK_bm)
79  {
80  // Slave send NACK instead of ACK - abort transmission
81  __I2CM_ErrorHandler(Device);
82  }
83  else
84  {
85  // Send the start address
86  if(__I2CM_Messages[Device].Status == I2C_MASTER_REGISTER)
87  {
88  __I2CM_Messages[Device].Device->MASTER.DATA = __I2CM_Messages[Device].Register;
89 
90  if(__I2CM_Messages[Device].BytesToRead > 0x00)
91  {
92  __I2CM_Messages[Device].Status = I2C_MASTER_ADDRESS;
93  }
94  else
95  {
96  __I2CM_Messages[Device].Status = I2C_MASTER_WRITE;
97  }
98  }
99  // Send repeated start after the address
100  else if(__I2CM_Messages[Device].Status == I2C_MASTER_ADDRESS)
101  {
102  __I2CM_Messages[Device].Device->MASTER.ADDR = I2C_READ(__I2CM_Messages[Device].DeviceAddress);
103  __I2CM_Messages[Device].Status = I2C_MASTER_READ;
104  }
105  // Send the data
106  else if(__I2CM_Messages[Device].Status == I2C_MASTER_WRITE)
107  {
108  if(__I2CM_Messages[Device].IndexWrite < __I2CM_Messages[Device].BytesToWrite)
109  {
110  __I2CM_Messages[Device].Device->MASTER.DATA = __I2CM_Messages[Device].BufferWrite[__I2CM_Messages[Device].IndexWrite++];
111  }
112  else
113  {
114  __I2CM_Messages[Device].Device->MASTER.CTRLC = TWI_MASTER_CMD_STOP_gc;
115  __I2CM_Messages[Device].Status = I2C_MASTER_SEND;
116 
117  if(__I2C_Callbacks[Device].TxCompleteInterrupt != NULL)
118  {
119  __I2C_Callbacks[Device].TxCompleteInterrupt();
120  }
121  }
122  }
123  }
124 }

129 static void __I2CM_ReadHandler(const uint8_t Device)
130 {
131  if(__I2CM_Messages[Device].Status == I2C_MASTER_READ)
132  {
133  // Check for buffer overflow
134  if(__I2CM_Messages[Device].IndexRead < TWI_BUFFER_SIZE)
135  {
136  __I2CM_Messages[Device].BufferRead[__I2CM_Messages[Device].IndexRead++] = __I2CM_Messages[Device].Device->MASTER.DATA;
137  }
138  else
139  {
140  __I2CM_Messages[Device].Device->MASTER.CTRLC = TWI_MASTER_ACKACT_bm | TWI_MASTER_CMD_STOP_gc;
141  __I2CM_Messages[Device].Status = I2C_MASTER_BUFFEROVERFLOW;
142  }
143 
144  if(__I2CM_Messages[Device].IndexRead < __I2CM_Messages[Device].BytesToRead)
145  {
146  __I2CM_Messages[Device].Device->MASTER.CTRLC = TWI_MASTER_CMD_RECVTRANS_gc;
147  }
148  else
149  {
150  __I2CM_Messages[Device].Device->MASTER.CTRLC = TWI_MASTER_ACKACT_bm | TWI_MASTER_CMD_STOP_gc;
151  __I2CM_Messages[Device].Status = I2C_MASTER_RECEIVED;
152 
153  if(__I2C_Callbacks[Device].RxCompleteInterrupt != NULL)
154  {
155  __I2C_Callbacks[Device].RxCompleteInterrupt();
156  }
157  }
158  }
159  else
160  {
161  __I2CM_ErrorHandler(Device);
162  }
163 }

168 static void __I2CM_InterruptHandler(const uint8_t Device)
169 {
170  uint8_t Status = I2CM_ReadStatus(__I2CM_Messages[Device].Device);
171 
172  // Check the interface status
173  if(Status & TWI_MASTER_ARBLOST_bm)
174  {
175  __I2CM_ArbitrationLostHandler(Device);
176  }
177  else if(Status & TWI_MASTER_WIF_bm)
178  {
179  __I2CM_WriteHandler(Device);
180  }
181  else if(Status & TWI_MASTER_RIF_bm)
182  {
183  __I2CM_ReadHandler(Device);
184  }
185  else
186  {
187  __I2CM_ErrorHandler(Device);
188  }
189 }

191 void I2CM_EnableInterruptSupport(TWI_t* Device, const Interrupt_Level_t Level)
192 {
193  Device->MASTER.CTRLA |= TWI_MASTER_RIEN_bm | TWI_MASTER_WIEN_bm;
194  Device->MASTER.CTRLA |= (Device->SLAVE.CTRLA & (~(0x03 << 0x06))) | (Level << 0x06);
195 }

233 static void __I2CS_DataHandler(const uint8_t Device)
234 {
235  uint8_t Status = __I2CS_Buffer[Device].Device->SLAVE.STATUS;
236 
237  // Transmit data to the master (Read from master)
238  if(Status & TWI_SLAVE_DIR_bm)
239  {
240  // Check if NACK
241  if(__I2CS_Buffer[Device].Device->SLAVE.STATUS & TWI_SLAVE_RXACK_bm)
242  {
243  __I2CS_Buffer[Device].Device->SLAVE.CTRLB = TWI_SLAVE_CMD_COMPTRANS_gc;
244  }
245  else
246  {
247  __I2CS_Buffer[Device].BytesSend++;
248  __I2CS_Buffer[Device].Device->SLAVE.DATA = __I2CS_Buffer[Device].Buffer[__I2CS_Buffer[Device].IndexRead++];
249 
250  if(__I2CS_Buffer[Device].IndexRead > TWI_BUFFER_SIZE)
251  {
252  __I2CS_Buffer[Device].IndexRead = 0x00;
253  __I2CS_Buffer[Device].Status = I2C_SLAVE_BUFFEROVERFLOW;
254  }
255 
256  __I2CS_Buffer[Device].Device->SLAVE.CTRLB |= TWI_SLAVE_CMD_RESPONSE_gc;
257  }
258  }
259  // Receive data from master (Write from master)
260  else
261  {
262  __I2CS_Buffer[Device].BytesReceived++;
263  __I2CS_Buffer[Device].Buffer[__I2CS_Buffer[Device].IndexWrite++] = __I2CS_Buffer[Device].Device->SLAVE.DATA;
264 
265  if(__I2CS_Buffer[Device].IndexWrite > TWI_BUFFER_SIZE)
266  {
267  __I2CS_Buffer[Device].Device->SLAVE.CTRLB = TWI_SLAVE_ACKACT_bm;
268  __I2CS_Buffer[Device].Status = I2C_SLAVE_BUFFEROVERFLOW;
269  }
270 
271  __I2CS_Buffer[Device].Device->SLAVE.CTRLB |= TWI_SLAVE_CMD_RESPONSE_gc;
272  }
273 }

278 static void __I2CS_InterruptHandler(const uint8_t Device)
279 {
280  uint8_t Status = I2CS_ReadStatus(__I2CS_Buffer[Device].Device);
281 
282  if(Status & TWI_SLAVE_BUSERR_bm)
283  {
284  __I2CS_ErrorHandler(Device);
285  }
286  else if(Status & TWI_SLAVE_COLL_bm)
287  {
288  __I2CS_ErrorHandler(Device);
289  }
290  else if((Status & TWI_SLAVE_APIF_bm) && (Status & TWI_SLAVE_AP_bm))
291  {
292  __I2CS_AddressHandler(Device);
293  }
294  else if(Status & TWI_SLAVE_APIF_bm)
295  {
296  __I2CS_StopHandler(Device);
297  }
298  else if(Status & TWI_SLAVE_DIF_bm)
299  {
300  __I2CS_DataHandler(Device);
301  }
302  else
303  {
304  __I2CS_ErrorHandler(Device);
305  }
306 }
