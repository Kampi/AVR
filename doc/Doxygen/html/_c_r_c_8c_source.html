39 static uint32_t CRC_GetChecksum(void)
40 {
41  uint32_t Checksum = 0x00;
42 
43  while((CRC.STATUS & CRC_BUSY_bm));
44 
45  if(CRC.CTRL & CRC_CRC32_bm)
46  {
47  Checksum = ((uint32_t)CRC.CHECKSUM0 & 0x000000FF);
48  Checksum |= (((uint32_t)CRC.CHECKSUM1 << 0x08) & 0x0000FF00);
49  Checksum |= (((uint32_t)CRC.CHECKSUM2 << 0x10) & 0x00FF0000);
50  Checksum |= (((uint32_t)CRC.CHECKSUM3 << 0x18) & 0xFF000000);
51  }
52  else
53  {
54  Checksum = ((uint16_t)CRC.CHECKSUM0 & 0x00FF);
55  Checksum |= (((uint16_t)CRC.CHECKSUM1 << 0x08) & 0xFF00);
56  }
57 
58  return Checksum;
59 }

64 static void CRC_WritePolynomial(const uint32_t Polynomial)
65 {
66  CRC_Disable();
67 
68  CRC.CHECKSUM0 = Polynomial & 0x000000FF;
69  CRC.CHECKSUM1 = (Polynomial & 0x0000FF00) >> 0x08;
70  CRC.CHECKSUM2 = (Polynomial & 0x00FF0000) >> 0x10;
71  CRC.CHECKSUM3 = (Polynomial & 0xFF000000) >> 0x18;
