44 static Endpoint_CS_ErrorCode_t USBStream_CheckControlEndpoint(void)
45 {
46  // Cancel transmission if the device got disconnected
47  if(__DeviceState == USB_STATE_UNATTACHED)
48  {
49  return ENDPOINT_CS_DISCONNECT;
50  }
51  // Cancel transmission if the device enter suspend mode
52  else if(__DeviceState == USB_STATE_SUSPEND)
53  {
54  return ENDPOINT_CS_SUSPEND;
55  }
56  // Cancel transmission if host sends an SETUP package
57  else if(Endpoint_SETUPReceived())
58  {
59  return ENDPOINT_CS_ABORT_FROM_HOST;
60  }
61 
62  return ENDPOINT_CS_NO_ERROR;
63 }

68 static Endpoint_DS_ErrorCode_t USBStream_WaitUntilReady(void)
69 {
70  uint8_t Timeout = 100;
71 
72  // Get the start frame

75  while(1)

92  }
93

120 Endpoint_CS_ErrorCode_t USBStream_INControl(const void* Buffer, const uint16_t Length)
121 {
122  uint8_t* Buffer_Temp = (uint8_t*)Buffer;
123  uint16_t Length_Temp = Length;
124 
125  if(Length > __ControlRequest.wLength)
126  {
127  Length_Temp = __ControlRequest.wLength;
128  }
129 
130  while(1)
131  {
132  Endpoint_CS_ErrorCode_t ErrorCode = USBStream_CheckControlEndpoint();
133  if(ErrorCode != ENDPOINT_CS_NO_ERROR)
134  {
135  return ErrorCode;
136  }
137  // Cancel transmission if an OUT request is received
138  else if(Endpoint_IsOUTFull())
139  {
140  break;
141  }
142 
143  if(Endpoint_IsINReady())
144  {
145  uint16_t BytesInEndpoint = Endpoint_GetBytes();
146 
147  while(Length_Temp && (BytesInEndpoint < ENDPOINT_CONTROL_SIZE))
148  {
149  Endpoint_WriteByte(pgm_read_byte(Buffer_Temp++));
150  Length_Temp--;
151  BytesInEndpoint++;
152  }
153 
154  // Start the transmission of the data
155  Endpoint_ClearIN();
156 
157  // Leave the loop when all bytes are send
158  if(Length_Temp == 0x00)
159  {
160  break;
161  }
162  }
163  }
164 
165  // Handle the STATUS stage (OUT transaction)
166  while(!(Endpoint_IsOUTFull()))
167  {
168  Endpoint_CS_ErrorCode_t ErrorCode = USBStream_CheckControlEndpoint();
169  if(ErrorCode != ENDPOINT_CS_NO_ERROR)
170  {
171  return ErrorCode;
172  }
173  }
174 
175  return ENDPOINT_CS_NO_ERROR;
176 }

178 uint8_t USBStream_INData(const void* Buffer, uint16_t Length, uint16_t* BytesProcessed)
179 {
180  uint8_t* Buffer_Temp = ((uint8_t*)Buffer);
181  uint16_t BytesInTransfer = 0x00;
182  Endpoint_DS_ErrorCode_t ErrorCode = ENDPOINT_DS_NO_ERROR;
183 
184  // Wait for the endpoint to become ready

187  {

190 
191  if(BytesProcessed != NULL)
192  {
193  Length -= *BytesProcessed;
194  Buffer_Temp += *BytesProcessed;
195  }
196 
197  while(Length)
198  {
199  // Endpoint is not empty
200  if(!(Endpoint_IsEmpty()))
201  {
202  Endpoint_ClearIN();
203 
204  if(BytesProcessed != NULL)
205  {
206  *BytesProcessed += BytesInTransfer;
207  return ENDPOINT_DS_INCOMPLETE;
208  }
209 
210  ErrorCode = USBStream_WaitUntilReady();
211  if(ErrorCode != ENDPOINT_DS_NO_ERROR)
212  {
213  return ErrorCode;
214  }
215  }
216  // Send data if the endpoint is empty
217  else
218  {
219  Endpoint_WriteByte(*Buffer_Temp++);
220  Length--;
221  BytesInTransfer++;
222  }
223  }
224 
225  return ENDPOINT_DS_NO_ERROR;
