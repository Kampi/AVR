49 void I2C_InstallCallback(const I2C_InterruptConfig_t* Config)
50 {
51  uint8_t ID = 0x00;
52 
53  if(Config->Device == &TWIC)
54  {
55  ID = TWIC_ID;
56  }
57  else if(Config->Device == &TWIE)
58  {
59  ID = TWIE_ID;
60  }
61 
62  if(Config->Source & I2C_TXCOMPLETE_INTERRUPT)
63  {
64  __I2C_Callbacks[ID].TxCompleteInterrupt = Config->Callback;
65  }
66 
67  if(Config->Source & I2C_RXCOMPLETE_INTERRUPT)
68  {
69  __I2C_Callbacks[ID].RxCompleteInterrupt = Config->Callback;
70  }
71 
72  if(Config->Source & I2C_ERROR_INTERRUPT)
73  {
74  __I2C_Callbacks[ID].ErrorInterrupt = Config->Callback;
75  }
76 
77  if(Config->Source & I2C_DATA_RDY_INTERRUPT)
78  {
79  __I2C_Callbacks[ID].DataRdyInterrupt = Config->Callback;
80  }
81 
82  if((Config->Source & I2C_TXCOMPLETE_INTERRUPT) || (Config->Source & I2C_RXCOMPLETE_INTERRUPT) || (Config->Source & I2C_ERROR_INTERRUPT))
83  {
84  Config->Device->MASTER.CTRLA = (Config->Device->MASTER.CTRLA & (~(0x03 << 0x06))) | (Config->InterruptLevel << 0x06);
85  }
86  else
87  {
88  Config->Device->SLAVE.CTRLA = (Config->Device->SLAVE.CTRLA & (~(0x03 << 0x06))) | (Config->InterruptLevel << 0x06);
89  }
90 }
