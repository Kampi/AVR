64 static inline void USART_InterruptHandler(const uint8_t Device, const USART_CallbackType_t Callback)
65 {
66  DisableGlobalInterrupts();
67 
68  if(Callback == USART_DRE_INTERRUPT)
69  {
70  if (USART_Callbacks[Device].EmptyCallback)
71  {
72  USART_Callbacks[Device].EmptyCallback();
73  }
74 
75  if(!(RingBuffer_IsEmpty(&USART_TxBuffer[0])))
76  {
77  UDR = RingBuffer_Load(&USART_TxBuffer[0]);
78  }
79  else
80  {
81  UCSRB &= ~(0x01 << UDRIE);
82  }
83  }
84  else if(Callback == USART_TXC_INTERRUPT)
85  {
86  if(USART_Callbacks[Device].TxCallback != NULL)
87  {
88  USART_Callbacks[Device].TxCallback();
89  }
90  }
91  else if(Callback == USART_RXC_INTERRUPT)
92  {
93  uint8_t ReceivedByte = UDR;
94 
95  // Echo message if enabled
96  if(USART_Echo == TRUE)
97  {
98  UDR = ReceivedByte;
99  }
100 
101  // Save the data in the buffer
102  RingBuffer_Save(&USART_RxBuffer[0], ReceivedByte);
103 
104  if(USART_Callbacks[Device].RxCallback != NULL)
105  {
106  USART_Callbacks[Device].RxCallback();
107  }
108  }
109 
110  EnableGlobalInterrupts();
111 }

113 void USART_Init(USART_Config_t* Config)
114 {
115  USART_SwitchEcho(USART_DEFAULT_ECHO);
116 
117  RingBuffer_Init(&USART_RxBuffer[0], RxData, USART_BUFFER_SIZE);
118  RingBuffer_Init(&USART_TxBuffer[0], TxData, USART_BUFFER_SIZE);
119 
120  USART_SetBaudrate(Config->Baudrate, SysClock_GetCPUClock(), Config->DoubleSpeed);
121  USART_SetDirection(Config->Direction);
122 
123  if(Config->Size == USART_SIZE_9)
124  {
125  UCSRB |= (0x01 << UCSZ2);
126  }
127 
128  // Config the USART interface
129  UCSRC = (0x01 << URSEL) | (Config->Mode << UMSEL) | (Config->Stop << USBS) | (Config->Parity << UPM0) | ((Config->Size & 0x03) << UCSZ0) | Config->ClockPolarity;
130 }

258 void USART_SetBaudrate(const uint32_t Baudrate, const uint32_t Clock, const Bool_t DoubleSpeed)
259 {
260  uint16_t BaudValue = ((Clock << DoubleSpeed) / (Baudrate << 0x04)) - 0x01;
261 
262  UBRRH &= ~(0x01 << URSEL);
263  UBRRH = (BaudValue >> 0x08);
264  UBRRL = (BaudValue & 0xFF);
265  UCSRA = (UCSRA & ~(0x01 << U2X)) | (DoubleSpeed << U2X);
266 }
