50 static void __AC_InterruptHandler(const uint8_t Device, const AC_CallbackType_t Type)
51 {
52  if(Type == AC_COMP0_INTERRUPT)
53  {
54  if(AC_Callbacks[Device].AC0)
55  {
56  AC_Callbacks[Device].AC0();
57  }
58  }
59  else if(Type == AC_COMP1_INTERRUPT)
60  {
61  if(AC_Callbacks[Device].AC1)
62  {
63  AC_Callbacks[Device].AC1();
64  }
65  }
66  else
67  {
68  if(AC_Callbacks[Device].Window)
69  {
70  AC_Callbacks[Device].Window();
71  }
72  }
73 }

252 {

254  {
255  Config->Device->AC0CTRL = (Config->Device->AC0CTRL & (~(0x03 << 0x06))) | (Config->Mode << 0x06);
256  Config->Device->AC0CTRL = (Config->Device->AC0CTRL & (~(0x03 << 0x04))) | (Config->InterruptLevel << 0x04);
257 
258  if(Config->Device == &ACB)
259  {
260  AC_Callbacks[ACB_ID].AC0 = Config->Callback;
261  }
262  else
263  {
264  AC_Callbacks[ACA_ID].AC0 = Config->Callback;
265  }
266 
267  Config->Device->STATUS |= AC_AC0IF_bm;
268  }
269 
270  if(Config->Source & AC_COMP1_INTERRUPT)
271  {
272  Config->Device->AC1CTRL = (Config->Device->AC1CTRL & (~(0x03 << 0x06))) | (Config->Mode << 0x06);
273  Config->Device->AC1CTRL = (Config->Device->AC1CTRL & (~(0x03 << 0x04))) | (Config->InterruptLevel << 0x04);
274 
275  if(Config->Device == &ACB)
276  {
277  AC_Callbacks[ACA_ID].AC1 = Config->Callback;
278  }
279  else
280  {
281  AC_Callbacks[ACA_ID].AC1 = Config->Callback;
282  }
283 
284  Config->Device->STATUS |= AC_AC0IF_bm;
285  }
286 
287  if(Config->Source & AC_WINDOW_INTERRUPT)
288  {
289  Config->Device->WINCTRL = (Config->Device->WINCTRL & (~(0x03 << 0x02))) | ((Config->Mode >> 0x01) << 0x02);
290  Config->Device->WINCTRL = (Config->Device->WINCTRL & (~0x03)) | Config->InterruptLevel;
