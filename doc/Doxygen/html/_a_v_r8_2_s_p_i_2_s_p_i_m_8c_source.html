57 void SPIM_Init(SPIM_Config_t* Config)
58 {
59  // Enable MOSI and SCK as output
60  GPIO_SetDirection(SPI_PORT, SPI_MOSI_PIN, GPIO_DIRECTION_OUT);
61  GPIO_SetDirection(SPI_PORT, SPI_SCK_PIN, GPIO_DIRECTION_OUT);
62 
63  // Enable MISO as input
64  GPIO_SetDirection(SPI_PORT, SPI_MISO_PIN, GPIO_DIRECTION_IN);
65 
66  // NOTE: The SS-Pin must held high in master mode, otherwise the MSTR-bit will be cleared
67  GPIO_Set(SPI_PORT, SPI_SS_PIN);
68 
69  SPI_SetDataOrder(Config->DataOrder);
70  SPIM_SetPrescaler(Config->Prescaler);
71  SPIM_SetMode(Config->Mode);
72  SPI_SetDeviceMode(SPI_MASTER);
73  SPI_Enable();
74 }
