47 static void __DMA_Channel_InterruptHandler(const uint8_t Channel)
48 {
49  uint8_t Status = DMA_ReadStatus();
50 
51  // Check for transaction complete interrupt
52  if(Status & (0x01 << Channel))
53  {
54  if(__DMA_Callbacks[Channel].TransactionComplete)
55  {
56  DMA_WriteStatus(Status | (0x01 << Channel));
57  __DMA_Callbacks[Channel].TransactionComplete(Channel);
58  }
59  }
60  // Check for error interrupt
61  else if((Status >> 0x04) & (0x01 << Channel))
62  {
63  if(__DMA_Callbacks[Channel].Error)
64  {
65  DMA_WriteStatus(Status | (0x0A << Channel));
66  __DMA_Callbacks[Channel].Error(Channel);
67  }
68  }
69 }

102 void DMA_Channel_ChangeInterruptLevel(DMA_CH_t* Channel, DMA_CallbackType_t Callback, Interrupt_Level_t InterruptLevel)
103 {
104  if(Callback && DMA_TRANSACTION_INTERRUPT)
105  {
106  Channel->CTRLB = (Channel->CTRLB & (~InterruptLevel)) | InterruptLevel;
107  }
108 
109  if(Callback && DMA_ERROR_INTERRUPT)
110  {
111  Channel->CTRLB = (Channel->CTRLB & (~(InterruptLevel << 0x02))) | (InterruptLevel << 0x02);
112  }

115 void DMA_Channel_InstallCallback(DMA_InterruptConfig_t* Config)
116 {
117  uint8_t Channel = 0x00;
118 
119  if(Config->Channel == &DMA.CH0)
120  {
121  Channel = 0x00;
122  }
123  #if(DMA_CHANNEL > 2)
124  else if(Config->Channel == &DMA.CH1)
125  {
126  Channel = 0x01;
127  }
128  #elif(DMA_CHANNEL > 2)
129  else if(Config->Channel == &DMA.CH2)
130  {
131  Channel = 0x02;
132  }
133  #elif(DMA_CHANNEL > 3)
134  else if(Config->Channel == &DMA.CH3)
135  {
136  Channel = 0x03;
137  }
138  #endif
139 
140  if(Config->Source & DMA_TRANSACTION_INTERRUPT)
141  {
142  Config->Channel->CTRLB |= (Config->Channel->CTRLB & (~Config->InterruptLevel)) | Config->InterruptLevel;
143  __DMA_Callbacks[Channel].TransactionComplete = Config->Callback;
144  }
145 
146  if(Config->Source & DMA_ERROR_INTERRUPT)
147  {
148  Config->Channel->CTRLB |= (Config->Channel->CTRLB & (~(Config->InterruptLevel << 0x02))) | (Config->InterruptLevel << 0x02);
149  __DMA_Callbacks[Channel].Error = Config->Callback;
150  }
151 }
