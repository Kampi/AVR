155  static inline void SysClock_SetClockSource(const ClockSource_t Source) __attribute__((always_inline));
156  static inline void SysClock_SetClockSource(const ClockSource_t Source)
157  {
158  uint8_t Flags = CPU_IRQSave();
159 
160  /*
161  * Use inline assembler, because otherwise the code will only run with at least
162  * optimization level -O1. This level is bad for debugging purposes. You can avoid
163  * this issue by using inline assembler.
164  */
165  asm volatile( "movw r30, %0" "\n\t"
166  "ldi r16, %2" "\n\t"
167  "out %3, r16" "\n\t"
168  "st Z, %1" "\n\t"
169  :: "r" (&CLK.CTRL),
170  "r" (Source & 0x07),
171  "M" (CCP_IOREG_gc),
172  "i" (&CCP)
173  : "r16", "r30"
174  );
175 
176  CPU_IRQRestore(Flags);
177  }
