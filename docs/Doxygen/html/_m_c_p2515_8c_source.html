375 static inline void MCP2515_Interrupthandler(void)
376 {
377  // Get all active interrupt flags
378  uint8_t Status = MCP2515_ReadRegister(MCP2515_REGISTER_CANINTF);
379 
380  // Message error interrupt
381  if(Status & (0x01 << MCP2515_IF_MER))
382  {
383  MCP2515_CLEAR_IF(0x01 << MCP2515_IF_MER);
384  }
385 
386  // Wakeup interrupt
387  if(Status & (0x01 << MCP2515_IF_WAK))
388  {
389  MCP2515_CLEAR_IF(0x01 << MCP2515_IF_WAK);
390 
391  if(__MCP2515_Callbacks.WakeCallback != NULL)
392  {
393  __MCP2515_Callbacks.WakeCallback();
394  }
395  }
396 
397  // Error interrupt
398  if(Status & (0x01 << MCP2515_IF_ERR))
399  {
400  MCP2515_CLEAR_IF(0x01 << MCP2515_IF_ERR);
401 
402  // Read the error flags
403  // ToDo: How to clear error flags in callback?
404  uint8_t ErrorFlags = MCP2515_ReadRegister(MCP2515_REGISTER_EFLG);
405 
406  // Error callback
407  if(__MCP2515_Callbacks.ErrorCallback != NULL)
408  {
409  __MCP2515_Callbacks.ErrorCallback(ErrorFlags);
410  }
411  }
412 
413  // Transmit buffer n full interrupt
414  if((Status & (0x01 << MCP2515_IF_TX2)) || (Status & (0x01 << MCP2515_IF_TX1)) || (Status & (0x01 << MCP2515_IF_TX0)))
415  {
416  MCP2515_CLEAR_IF((0x01 << MCP2515_IF_TX2) | (0x01 << MCP2515_IF_TX1) | (0x01 << MCP2515_IF_TX0));
417 
418  // Clear the transmit buffer full bits
419  __TxBufferFull &= ~((Status >> 0x02) & 0x07);
420  }
421 
422  // Receive buffer n full interrupt
423  if((Status & (0x01 << MCP2515_IF_RX1)) || (Status & (0x01 << MCP2515_IF_RX0)))
424  {
425  // General message callback for all messages
426  if(__MCP2515_Callbacks.RxCallback != NULL)
427  {
428  __MCP2515_Callbacks.RxCallback();
429  }
430  }
431 }

436 static void MCP2515_EnableInterrupts(const MCP2515_Config_t* Config)
437 {
438  // Initialize interrupt pin
439  GPIO_SetDirection(Config->Port, Config->Pin, GPIO_DIRECTION_IN);
440 
441  GPIO_InterruptConfig_t ConfigGPIO = {
442  .Channel = Config->Channel,
443  .Sense = GPIO_SENSE_LOWLEVEL,
444  #if(MCU_ARCH == MCU_ARCH_XMEGA)
445  .Pin = Config->Pin,
446  .Port = Config->Port,
447  .InterruptLevel = Config->InterruptLevel,
448  #endif
449  .Callback = MCP2515_Interrupthandler
450  };
451 
452  GPIO_InstallCallback(&ConfigGPIO);
453 
454  // Save the callbacks
455  if(Config->WakeCallback != NULL)
456  {
457  __MCP2515_Callbacks.WakeCallback = Config->WakeCallback;
458  }
459 
460  if(Config->TxCallback != NULL)
461  {
462  __MCP2515_Callbacks.TxCallback = Config->TxCallback;
463  }
464 
465  if(Config->RxCallback != NULL)
466  {
467  __MCP2515_Callbacks.RxCallback = Config->RxCallback;
468  }
469 
470  // Enable device interrupts
471  MCP2515_WriteRegister(MCP2515_REGISTER_CANINTE, (0x01 << MCP2515_IF_MER) | (0x01 << MCP2515_IF_WAK) | (0x01 << MCP2515_IF_ERR) | (0x01 << MCP2515_IF_TX2) | (0x01 << MCP2515_IF_TX1) | (0x01 << MCP2515_IF_TX0) | (0x01 << MCP2515_IF_RX1) | (0x01 << MCP2515_IF_RX0));
472 }

474 void MCP2515_Init(SPIM_Config_t* Config, const MCP2515_Config_t* DeviceConfig)
475 {
476  __TxBufferFull = 0x00;
477 
478  // Initialize the CS Pin
479  GPIO_SetDirection(GET_PERIPHERAL(MCP2515_SS), GET_INDEX(MCP2515_SS), GPIO_DIRECTION_OUT);
480  GPIO_Set(GET_PERIPHERAL(MCP2515_SS), GET_INDEX(MCP2515_SS));
481 
482  // Initialize the external reset I/O
483  #if(defined MCP2515_USE_EXT_RESET)
484  GPIO_SetDirection(GET_PERIPHERAL(MCP2515_EXT_RESET), GET_INDEX(MCP2515_EXT_RESET), GPIO_DIRECTION_OUT);
485  GPIO_Set(GET_PERIPHERAL(MCP2515_EXT_RESET), GET_INDEX(MCP2515_EXT_RESET));
486 
487  // Wait a bit
488  for(uint8_t i = 0x00; i < 0xFF; i++);
489  #endif
490 
491  // Initialize the external request to send I/O
492  #if(defined MCP2515_USE_EXT_TX)
493  GPIO_SetDirection(GET_PERIPHERAL(MCP2515_EXT_TX0), GET_INDEX(MCP2515_EXT_TX0), GPIO_DIRECTION_OUT);
494  GPIO_Set(GET_PERIPHERAL(MCP2515_EXT_TX0), GET_INDEX(MCP2515_EXT_TX0));
495  GPIO_SetDirection(GET_PERIPHERAL(MCP2515_EXT_TX1), GET_INDEX(MCP2515_EXT_TX1), GPIO_DIRECTION_OUT);
496  GPIO_Set(GET_PERIPHERAL(MCP2515_EXT_TX1), GET_INDEX(MCP2515_EXT_TX1));
497  GPIO_SetDirection(GET_PERIPHERAL(MCP2515_EXT_TX2), GET_INDEX(MCP2515_EXT_TX2), GPIO_DIRECTION_OUT);
498  GPIO_Set(GET_PERIPHERAL(MCP2515_EXT_TX2), GET_INDEX(MCP2515_EXT_TX2));
499 
500  // Enable TxnRTS pins in the device
501  MCP2515_WriteRegister(MCP2515_REGISTER_TXRTSCTRL, (0x01 << MCP2515_B2RTSM) | (0x01 << MCP2515_B1RTSM) | (0x01 << MCP2515_B0RTSM));
502  #endif
503 
504  // Initialize the SPI-Interface
505  if(Config != NULL)
506  {
507  MCP2515_SPI_INIT(Config);
508  }
509 
510  // Initialize MCP2515
511  MCP2515_Reset();
512 
513  // Setup Interrupts
514  MCP2515_EnableInterrupts(DeviceConfig);
515 
516  // Switch the device into configuration mode
517  MCP2515_SetDeviceMode(MCP2515_CONFIG_MODE);
518 
519  // Set the double shot bit
520  MCP2515_BitModify(MCP2515_REGISTER_CANCTRL, (0x01 << MCP2515_OSM), (DeviceConfig->EnableOneShot << MCP2515_OSM));
521 
522  // Set the rollover bit
523  MCP2515_BitModify(MCP2515_REGISTER_RXB0CTRL, (0x01 << MCP2515_BUKT), (DeviceConfig->EnableRollover << MCP2515_BUKT));
524 
525  // Use default timing
526  if(DeviceConfig->pTiming == NULL)
527  {
528  MCP2515_WriteRegister(MCP2515_REGISTER_CNF1, 0x03);
529  MCP2515_WriteRegister(MCP2515_REGISTER_CNF2, 0xA0);
530  MCP2515_WriteRegister(MCP2515_REGISTER_CNF3, 0x02);
531  }
532  else
533  {
534  MCP2515_ConfigureTiming(DeviceConfig->pTiming);
535  }
536 
537  // Disable all mask and filter
538  MCP2515_BitModify(MCP2515_REGISTER_RXB0CTRL, (0x01 << MCP2515_RXM1) | (0x01 << MCP2515_RXM0), (0x01 << MCP2515_RXM1) | (0x01 << MCP2515_RXM0));
539  MCP2515_BitModify(MCP2515_REGISTER_RXB1CTRL, (0x01 << MCP2515_RXM1) | (0x01 << MCP2515_RXM0), (0x01 << MCP2515_RXM1) | (0x01 << MCP2515_RXM0));
540 
541  // Clear all interrupt flags
542  MCP2515_WriteRegister(MCP2515_REGISTER_CANINTF, 0x00);
543 
544  // Switch the device mode back to normal mode
545  MCP2515_SetDeviceMode(MCP2515_NORMAL_MODE);
546 
547  // Switch the device into loop back mode
548  if(DeviceConfig->EnableLoopBack)
549  {
550  MCP2515_SetDeviceMode(MCP2515_LOOPBACK_MODE);
551  }
552 }

655 void MCP2515_ConfigFilter(const MCP2515_FilterConfig_t* Config)
656 {
657  uint8_t IdentifierBaseAddress = 0x00;
658 
659  // Get the current device mode
660  MCP2515_DeviceMode_t Mode = MCP2515_GetDeviceMode();
661 
662  // Change mode to configuration mode to change the filter setup
663  MCP2515_SetDeviceMode(MCP2515_CONFIG_MODE);
664 
665  uint8_t Mask_Low = (Config->Mask & 0x07) << 0x05;
666  uint8_t Mask_High = Config->Mask >> 0x03;
667 
668  switch(Config->Filter)
669  {
670  case MCP2515_RXF0:
671  {
672  IdentifierBaseAddress = MCP2515_REGISTER_RXF0SIDH;
673  break;
674  }
675  case MCP2515_RXF1:
676  {
677  IdentifierBaseAddress = MCP2515_REGISTER_RXF1SIDH;
678  break;
679  }
680  case MCP2515_RXF2:
681  {
682  IdentifierBaseAddress = MCP2515_REGISTER_RXF2SIDH;
683  break;
684  }
685  case MCP2515_RXF3:
686  {
687  IdentifierBaseAddress = MCP2515_REGISTER_RXF3SIDH;
688  break;
689  }
690  case MCP2515_RXF4:
691  {
692  IdentifierBaseAddress = MCP2515_REGISTER_RXF4SIDH;
693  break;
694  }
695  case MCP2515_RXF5:
696  {
697  IdentifierBaseAddress = MCP2515_REGISTER_RXF5SIDH;
698  break;
699  }
700  }
701 
702  // Set the identifier
703  if(Config->ID > 0x00)
704  {
705  uint8_t Identifier_Low = (Config->ID & 0x07) << 0x05;
706  uint8_t Identifier_High = Config->ID >> 0x03;
707 
708  MCP2515_WriteRegister(IdentifierBaseAddress, Identifier_High);
709  MCP2515_WriteRegister(IdentifierBaseAddress + 0x01, Identifier_Low);
710  }
711 
712  if((Config->Filter == MCP2515_RXF0) || (Config->Filter == MCP2515_RXF1))
713  {
714  if(!Config->UseExtended)
715  {
716  MCP2515_WriteRegister(MCP2515_REGISTER_RXM0SIDH, Mask_High);
717  MCP2515_WriteRegister(MCP2515_REGISTER_RXM0SIDL, Mask_Low);
718  }
719  }
720  else if((Config->Filter == MCP2515_RXF2) | (Config->Filter == MCP2515_RXF3) | (Config->Filter == MCP2515_RXF4) | (Config->Filter == MCP2515_RXF5))
721  {
722  if(!Config->UseExtended)
723  {
724  MCP2515_WriteRegister(MCP2515_REGISTER_RXM1SIDH, Mask_High);
725  MCP2515_WriteRegister(MCP2515_REGISTER_RXM1SIDL, Mask_Low);
726  }
727  }
728 
729  // Restore device mode
730  MCP2515_SetDeviceMode(Mode);
731 }

776 void MCP2515_ReadMessage(CAN_Message_t* Message)
777 {
778  uint8_t ID_High = 0x00;
779  uint8_t ID_Low = 0x00;
780  uint8_t Length = 0x00;
781  uint8_t ReceiveBuffer = 0x00;
782 
783  // Get the receive status
784  uint8_t RxStatus = MCP2515_ReadRxStatus();
785 
786  // Return when no message received
787  if(!(RxStatus & ((0x01 << MCP2515_MSG_RX_1) | (0x01 << MCP2515_MSG_RX_0))))
788  {
789  return;
790  }
791  else if(RxStatus & (0x01 << MCP2515_MSG_RX_1))
792  {
793  ReceiveBuffer = MCP2515_CMD_READ_RX(MCP2515_RX1);
794  }
795  else
796  {
797  ReceiveBuffer = MCP2515_CMD_READ_RX(MCP2515_RX0);
798  }
799 
800  MCP2515_SPI_CHIP_SELECT();
801  MCP2515_SPI_TRANSMIT(ReceiveBuffer);
802 
803  // Get the message ID
804  ID_High = (MCP2515_SPI_TRANSMIT(0xFF)) << 0x03;
805  ID_Low = (MCP2515_SPI_TRANSMIT(0xFF)) >> 0x05;
806  Message->ID = ID_High | ID_Low;
807 
808  // Read two empty bytes
809  MCP2515_SPI_TRANSMIT(0xFF);
810  MCP2515_SPI_TRANSMIT(0xFF);
811 
812  // Get the message length
813  Length = MCP2515_SPI_TRANSMIT(0xFF);
814 
815  // Check message type
816  uint8_t MessageType = (RxStatus & 0x18) >> 0x03;
817 
818  switch(MessageType)
819  {
820  case(MCP2515_STANDARD_DATA):
821  {
822  Message->Length = Length;
823  Message->Type = MessageType;
824 
825  for(uint8_t i = 0; i < Message->Length; i++)
826  {
827  *(Message->pData++) = MCP2515_SPI_TRANSMIT(0xFF);
828  }
829 
830  break;
831  }
832  case(MCP2515_STANDARD_REMOTE):
833  {
834  // Remove the remote bit
835  Message->Length = Length - 0x40;
836 
837  Message->Type = MessageType;
838 
839  break;
840  }
841  }
842 
843  MCP2515_SPI_CHIP_DESELECT();
844 }

846 MCP2515_ErrorCode_t MCP2515_PrepareMessage(CAN_Message_t* Message, const MCP2515_MessagePriority_t Priority)
847 {
848  if(Message->ID > 0x7FF)
849  {
850  return MCP2515_INVALID_IDENTIFIER;
851  }
852 
853  // Check for an empty transmit buffer. First check the internal bitmap and then check
854  // the TXREQ bit in the second step
855  uint8_t Buffer = 0x00;
856  if(!(__TxBufferFull & MCP2515_TX0))
857  {
858  if(!(MCP2515_ReadRegister(MCP2515_REGISTER_TXB0CTRL) & MCP2515_TXREQ))
859  {
860  Buffer = MCP2515_REGISTER_TXB0CTRL;
861  }
862 
863  __TxBufferFull |= MCP2515_TX0;
864  }
865  else if(!(__TxBufferFull & MCP2515_TX1))
866  {
867  if(!(MCP2515_ReadRegister(MCP2515_REGISTER_TXB1CTRL) & MCP2515_TXREQ))
868  {
869  Buffer = MCP2515_REGISTER_TXB1CTRL;
870  }
871 
872  __TxBufferFull |= MCP2515_TX1;
873  }
874  else if(!(__TxBufferFull & MCP2515_TX2))
875  {
876  if(!(MCP2515_ReadRegister(MCP2515_REGISTER_TXB2CTRL) & MCP2515_TXREQ))
877  {
878  Buffer = MCP2515_REGISTER_TXB2CTRL;
879  }
880 
881  __TxBufferFull |= MCP2515_TX2;
882  }
883 
884  // All Tx buffers are full - cancel transmission
885  if(Buffer == 0x00)
886  {
887  return MCP2515_TX_BUFFER_FULL;
888  }
889 
890  // Delete TXREQ Bit
891  MCP2515_BitModify(Buffer, (0x01 << MCP2515_TXREQ), 0x00);
892 
893  // Set the message id
894  MCP2515_WriteRegister(Buffer + 0x02, (Message->ID & 0x07) << 0x05);
895  MCP2515_WriteRegister(Buffer + 0x01, Message->ID >> 0x03);
896 
897  // Set the message priority
898  MCP2515_BitModify(Buffer, 0x03, Priority & 0x03);
899 
900  switch(Message->Type)
901  {
902  case MCP2515_STANDARD_DATA:
903  {
904  MCP2515_WriteRegister(Buffer + 0x05, (Message->Length & 0x0F));
905 
906  for(uint8_t DataByte = 0x00; DataByte < MCP2515_MAX_DATABYTES; DataByte++)
907  {
908  MCP2515_WriteRegister(Buffer + 0x06 + DataByte, *(Message->pData++));
909  }
910 
911  break;
912  }
913  case MCP2515_STANDARD_REMOTE:
914  {
915  MCP2515_WriteRegister(Buffer + 0x05, (Message->Length & 0x0F) + (0x01 << MCP2515_RTR));
916 
917  break;
918  }
919  case MCP2515_EXTENDED_DATA:
920  {
921  break;
922  }
923  case MCP2515_EXTENDED_REMOTE:
924  {
925  break;
926  }
927  }
928 
929  return MCP2515_NO_ERROR;
930 }
