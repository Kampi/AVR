154 void NVM_FlashRangeCRC(const uint32_t Start, const uint32_t Length)
155 {
156  // Load the start byte address in the NVM Address Register
157  NVM.ADDR0 = Start & 0xFF;
158  NVM.ADDR1 = (Start >> 0x08) & 0xFF;
159  NVM.ADDR2 = (Start >> 0x10) & 0xFF;
160 
161  // Load the end byte address in NVM Data Register
162  uint32_t EndAddr = Start + Length - 0x01;
163  NVM.DATA0 = EndAddr & 0xFF;
164  NVM.DATA1 = (EndAddr >> 0x08) & 0xFF;
165  NVM.DATA2 = (EndAddr >> 0x010) & 0xFF;
166 
167  NVM_ExecuteCommand(NVM_CMD_FLASH_RANGE_CRC_gc);
168 }
