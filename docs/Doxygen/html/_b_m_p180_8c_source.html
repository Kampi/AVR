162 const I2C_Error_t BMP180_MeasureTemperature(BMP180_DataPoint_t* DataPoint)
163 {
164  I2C_Error_t ErrorCode = I2C_NO_ERROR;
165  uint8_t Data[2] = {BMP180_REGISTER_CTRL, 0x00};
166 
167  if(DataPoint == NULL)
168  {
169  return I2C_INVALID_PARAM;
170  }
171 
172  // Start the temperature measurement
173  Data[1] = BMP180_CMD_TEMP_MEAS;
174  ErrorCode = BMP180_I2CM_WRITEBYTES(2, Data, TRUE);
175  if(ErrorCode != I2C_NO_ERROR)
176  {
177  return ErrorCode;
178  }
179 
180  // Wait for measurement end
181  do
182  {
183  ErrorCode = BMP180_I2CM_WRITEBYTE(Data[0], FALSE) | BMP180_I2CM_READBYTE(&Data[1], TRUE);
184  if(ErrorCode != I2C_NO_ERROR)
185  {
186  return ErrorCode;
187  }
188  } while(Data[1] & (0x01 << BMP180_START_CONV));
189 
190  // Read the temperature
191  ErrorCode = BMP180_I2CM_WRITEBYTE(BMP180_REGISTER_OUT_MSB, FALSE) | BMP180_I2CM_READBYTES(sizeof(Data), Data, TRUE);
192  if(ErrorCode != I2C_NO_ERROR)
193  {
194  return ErrorCode;
195  }
196 
197  DataPoint->Temperature = (Data[0] << 0x08) + Data[1];
198 
199  return ErrorCode;

242 const I2C_Error_t BMP180_SingleMeasurement(const BMP180_OSS_t OSS, const BMP180_CalibCoef_t* CalibCoef, BMP180_DataPoint_t* DataPoint)
243 {
244  uint32_t B4, B7;
245  int32_t X1, X2, X3, B3, B5, B6, p;
246 
247  I2C_Error_t ErrorCode = I2C_NO_ERROR;
248 
249  if((CalibCoef == NULL) || (DataPoint == NULL))
250  {
251  return I2C_INVALID_PARAM;
252  }
253 
254  ErrorCode = BMP180_MeasureTemperature(DataPoint) | BMP180_MeasurePressure(OSS, DataPoint);
255  if(ErrorCode != I2C_NO_ERROR)
256  {
257  return ErrorCode;
258  }
259 
260  // Calibrate the temperature (see the official datasheet for more information)
261  X1 = (((int32_t)DataPoint->Temperature - CalibCoef->AC6) * CalibCoef->AC5) >> 15;
262  X2 = ((int32_t)CalibCoef->MC << 11) / (X1 + CalibCoef->MD);
263  B5 = X1 + X2;
264  DataPoint->Temperature = ((float)B5 + 8) / 160;
265 
266  // Calibrate the pressure (see the official datasheet for more information)
267  B6 = B5 - 4000;
268  X1 = (CalibCoef->B2 * ((B6 * B6) >> 12)) >> 11;
269  X2 = (CalibCoef->AC2 * B6) >> 11;
270  X3 = X1 + X2;
271  B3 = (((((int32_t)CalibCoef->AC1 << 2) + X3) << OSS) + 2) >> 2;
272  X1 = (CalibCoef->AC3 * B6) >> 13;
273  X2 = CalibCoef->B1 * ((B6 * B6) >> 12) >> 16;
274  X3 = (X1 + X2 + 2) >> 2;
275  B4 = (CalibCoef->AC4 * (uint32_t)(X3 + 32768)) >> 15;
276  B7 = (DataPoint->Pressure - B3) * (50000 >> OSS);
277 
278  if(B7 < 0x80000000)
279  {
280  p = (B7 << 1) / B4;
281  }
282  else
283  {
284  p = (B7 / B4) << 1;
285  }
286 
287  X1 = p >> 8;
288  X1 *= X1;
289  X1 = (X1 * 3038) >> 16;
290  X2 = (-7357 * p) >> 16;
291  p = p + ((X1 + X2 + 3791) >> 4);
292 
293  DataPoint->Pressure = p;
294 
295  return ErrorCode;
296 }
