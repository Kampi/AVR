34 void DisplayManager_DrawPixel(const uint8_t x, const uint8_t y, const PixelMask_t Mask)
35 {
36  // Check if the coordinates are outside of the screen
37  if((x > (DISPLAYMANAGER_LCD_WIDTH - 1)) || (y > (DISPLAYMANAGER_LCD_HEIGHT - 1)))
38  {
39  return;
40  }
41 
42  // Get the byte for the chosen pixel
43  uint8_t Page = y / DISPLAYMANAGER_LCD_PIXEL_PER_BYTE;
44  uint8_t ByteMask = (0x01 << (y - (Page * DISPLAYMANAGER_LCD_PIXEL_PER_BYTE)));
45 
46  // Read the byte from the frame buffer
47  uint8_t Byte = DisplayManager_ReadByte(Page, x);
48 
49  // Set the pixel
50  if(Mask == PIXELMASK_SET)
51  {
52  Byte |= ByteMask;
53  }
54  else
55  {
56  Byte &= ~ByteMask;
57  }
58 
59  // Write the new value to the display
60  DisplayManager_WriteByte(Page, x, Byte);
61 }

66 void DisplayManager_DrawLine(const uint8_t x1, const uint8_t y1, const uint8_t x2, const uint8_t y2, const PixelMask_t Mask)
67 {
68  int8_t e = 0; // Error variable
69  int8_t incr_x = 1; // Increment direction x axis
70  int8_t incr_y = 1; // Increment direction y axis
71  int8_t pdx = 0; // Parallel step x axis
72  int8_t pdy = 0; // Parallel step y axis
73  int8_t ddx = 0; // Diagonal step x axis
74  int8_t ddy = 0; // Diagonal step y axis
75  int8_t ds = 0; // Delta fast direction
76  int8_t df = 0; // Delta slow direction
77  int8_t x = x1; // Current point x axis
78  int8_t y = y1; // Current point y axis
79 
80  // Get the distance between both points
81  int8_t dx = x2 - x1;
82  int8_t dy = y2 - y1;
83 
84  // Get increment direction of x
85  if(dx < 0)
86  {
87  incr_x = -1;
88  dx = -dx;
89  }
90 
91  // Get increment direction of y
92  if(dy < 0)
93  {
94  incr_y = -1;
95  dy = -dy;
96  }
97 
98  // Chose the biggest distance
99  if(dx > dy)
100  {
101  pdx = incr_x;
102  ddx = incr_x;
103  ddy = incr_y;
104  ds = dy;
105  df = dx;
106  }
107  else
108  {
109  pdy = incr_y;
110  ddx = incr_x;
111  ddy = incr_y;
112  ds = dx;
113  df = dy;
114  }
115 
116  e = df >> 0x01;
117  DisplayManager_DrawPixel(x, y, Mask);
118 
119  for(uint8_t i = 0x00; i < df; i++)
120  {
121  e -= ds;
122 
123  if(e < 0)
124  {
125  e += df;
126  x += ddx;
127  y += ddy;
128  }
129  else
130  {
131  x += pdx;
132  y += pdy;
133  }
134 
135  DisplayManager_DrawPixel(x, y, Mask);
136  }
137 }

194 void DisplayManager_DrawCircle(const uint8_t x, const uint8_t y, const uint8_t Radius, const FillOptions_t Fill, const PixelMask_t Mask)
195 {
196  // Check if the rectangle fits on the screen
197  if(((x + Radius) > (DISPLAYMANAGER_LCD_WIDTH - 1)) || ((y + Radius) > (DISPLAYMANAGER_LCD_HEIGHT - 1)))
198  {
199  return;
200  }
201 
202  // Skip drawing if radius is zero
203  if(Radius == 0)
204  {
205  DisplayManager_DrawPixel(x, y, Mask);
206  return;
207  }
208 
209  uint8_t OffsetX;
210  uint8_t OffsetY;
211  int16_t Error;
212 
213  OffsetX = 0;
214  OffsetY = Radius;
215  Error = 3 - (Radius << 0x01);
216 
217  while(OffsetX <= OffsetY)
218  {
219  if(Fill == FILL_NO)
220  {
221  DisplayManager_DrawPixel(x + OffsetY, y - OffsetX, Mask);
222  DisplayManager_DrawPixel(x + OffsetX, y - OffsetY, Mask);
223  DisplayManager_DrawPixel(x - OffsetX, y - OffsetY, Mask);
224  DisplayManager_DrawPixel(x - OffsetY, y - OffsetX, Mask);
225  DisplayManager_DrawPixel(x - OffsetY, y + OffsetX, Mask);
226  DisplayManager_DrawPixel(x - OffsetX, y + OffsetY, Mask);
227  DisplayManager_DrawPixel(x + OffsetX, y + OffsetY, Mask);
228  DisplayManager_DrawPixel(x + OffsetY, y + OffsetX, Mask);
229  }
230  else
231  {
232  DisplayManager_DrawVerticalLine(x + OffsetY, y - OffsetX, OffsetX + 1, 1, Mask);
233  DisplayManager_DrawVerticalLine(x + OffsetX, y - OffsetY, OffsetY + 1, 1, Mask);

240  }

376 void DisplayManager_DrawBitmap(const uint8_t x, const uint8_t y, const Bitmap_t* Bitmap)
377 {
378  uint8_t i = 0x00;
379  uint8_t Height = 0x00;
380 
381  if(((x + Bitmap->Width) > (DISPLAYMANAGER_LCD_WIDTH - 1)) || ((y + Bitmap->Height) > (DISPLAYMANAGER_LCD_HEIGHT - 1)))
382  {
383  return;
384  }
385 
386  while(Height < Bitmap->Height)
387  {
388  for(uint8_t j = 0x00; j < Bitmap->Width; j++)
389  {
390  uint8_t Byte = 0x00;
391 
392  if(Bitmap->Type == MEMORY_PROGMEM)
393  {
394  Byte = pgm_read_byte(Bitmap->Data.Flash + (i * Bitmap->Width) + j);
395  }
396  else if(Bitmap->Type == MEMORY_RAM)
397  {
398  Byte = *(Bitmap->Data.RAM + (i * Bitmap->Width) + j);
399  }
400 
401  for(uint8_t k = 0x00; k < 0x08; k++)
402  {
403  // Draw a pixel if the current LSB is set
404  if(Byte & 0x01)
405  {
406  DisplayManager_DrawPixel(x + j, y + i * DISPLAYMANAGER_LCD_PIXEL_PER_BYTE + k, PIXELMASK_SET);
407  }
408 
409  // Shift the mask
410  Byte = Byte >> 0x01;
411  }
412  }
413 
414  if((Height + DISPLAYMANAGER_LCD_PIXEL_PER_BYTE) < Bitmap->Height)
415  {
416  Height += DISPLAYMANAGER_LCD_PIXEL_PER_BYTE;
417  }
418  else
419  {
420  Height += Bitmap->Height - Height;
421  }
422 
423  i++;
424  }
425 }

427 uint8_t DisplayManager_DrawString(const uint8_t x, const uint8_t y, const char* String)
428 {
429  uint8_t x_temp = x;
430  uint8_t y_temp = y;
431 
432  while(*String)
433  {
434  // Get the character from string
435  uint8_t ASCII = *String++;
436 
437  // Check if character is valid
438  if((ASCII > 31) && (ASCII < 127))
439  {
440  // Get the font layout for the character
441  const uint8_t* Layout = FontTable[ASCII - 32];
442 
443  uint8_t Columns = *Layout++;
444 
445  // Add line break
446  if((x_temp + Columns) > DISPLAYMANAGER_LCD_WIDTH)
447  {
448  x_temp = 0x00;
449  y_temp += FONT_HEIGHT;
450  }
451 
452  // Loop over the columns layout (first item is the column count)
453  for(uint8_t i = 0x00; i < Columns; i++)
454  {
455  // Get the bit mask
456  uint8_t Mask = *Layout++;
457 
458  for(uint8_t j = 0x00; j < 0x08; j++)
459  {
460  // Draw a pixel if the current LSB is set
461  if(Mask & 0x01)
462  {
463  DisplayManager_DrawPixel(x_temp, y_temp + j, PIXELMASK_SET);
464  }
465 
466  // Shift the mask
467  Mask = Mask >> 0x01;
468  }
469 
470  // Next column
471  x_temp++;
472  }
473 
474  // Add one pixel spacing between the characters
475  x_temp++;
476  }
477  }
478 
479  return x_temp;
480 }
