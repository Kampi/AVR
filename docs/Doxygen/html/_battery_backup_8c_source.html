36 void BatteryBackup_Reset(void)
37 {
38  // Enable R/W access
39  VBAT.CTRL = VBAT_ACCEN_bm;
40 
41  // Perform reset
42  ATOMIC_BLOCK(ATOMIC_RESTORESTATE)
43  {
44  asm volatile( "movw r30, %0" "\n\t"
45  "ldi r16, %2" "\n\t"
46  "out %3, r16" "\n\t"
47  "st Z, %1" "\n\t"
48  :: "r" (&VBAT.CTRL),
49  "r" (VBAT_RESET_bm),
50  "M" (CCP_IOREG_gc),
51  "i" (&CCP)
52  : "r16", "r30", "r31"
53  );
54  }
55 }
