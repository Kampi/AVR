146 void Display_Init(SPIM_Config_t* Config)
147 {
148  // Set Data/Command pin as output in low state
149  GPIO_SetDirection(GET_PERIPHERAL(SSD1306_DATA), GET_INDEX(SSD1306_DATA), GPIO_DIRECTION_OUT);
150  GPIO_Set(GET_PERIPHERAL(SSD1306_DATA), GET_INDEX(SSD1306_DATA));
151 
152  // Set display reset as output in high state
153  GPIO_SetDirection(GET_PERIPHERAL(SSD1306_RESET), GET_INDEX(SSD1306_RESET), GPIO_DIRECTION_OUT);
154  GPIO_Set(GET_PERIPHERAL(SSD1306_RESET), GET_INDEX(SSD1306_RESET));
155 
156  // Set display SS as output in high state
157  GPIO_SetDirection(GET_PERIPHERAL(SSD1306_SS), GET_INDEX(SSD1306_SS), GPIO_DIRECTION_OUT);
158  GPIO_Set(GET_PERIPHERAL(SSD1306_SS), GET_INDEX(SSD1306_SS));
159 
160  Display_Reset();
161 
162  // Initialize SPI-Interface
163  if(Config != NULL)
164  {
165  SSD1306_SPIM_INIT(Config);
166  }
167 
168  SSD1306_WriteCommandBytes(DisplayInitialization, sizeof(DisplayInitialization));
169 }

185 void Display_SetColumn(const uint8_t Column)
186 {
187  // Use only the lower 4 bits, because the controller supports only 128 columns
188  uint8_t Temp = Column & 0x7F;
189  SSD1306_WriteCommand(SSD1306_CMD_HIGH_COL(Temp >> 0x04));
190  SSD1306_WriteCommand(SSD1306_CMD_LOW_COL(Temp & 0x0F));
191 }
