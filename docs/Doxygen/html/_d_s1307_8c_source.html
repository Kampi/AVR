253 const I2C_Error_t DS1307_SetHourMode(const HourMode_t Mode)
254 {
255  uint8_t Data = 0x00;
256  I2C_Error_t ErrorCode = I2C_NO_ERROR;
257 
258  ErrorCode = DS1307_I2CM_WRITEBYTE(DS1307_REGISTER_HOURS, FALSE) | DS1307_I2CM_READBYTE(&Data, TRUE);
259  if(ErrorCode != I2C_NO_ERROR)
260  {
261  return ErrorCode;
262  }
263 
264  // Change from 24-Hour mode to 12-Hour mode
265  if(Mode == MODE_12_HOUR)
266  {
267  Data = BCD2Dec(Data);
268 
269  // Check if AM or PM
270  if(Data > 12)
271  {
272  Data -= 12;
273  Data |= (0x01 << DS1307_MERIDIEM);
274  }
275  else
276  {
277  Data &= ~(0x01 << DS1307_MERIDIEM);
278  }
279 
280  Data = Dec2BCD(Data);
281 
282  // Set 12-Hour mode
283  Data |= (0x01 << DS1307_12_24);
284  }
285  // Change from 12-Hour mode to 24-Hour mode
286  else if(Mode == MODE_24_HOUR)
287  {
288  // Set 24-Hour mode
289  Data &= ~(0x01 << DS1307_12_24);
290 
291  if(Data & (0x01 << DS1307_MERIDIEM))
292  {
293  Data &= ~(0x01 << DS1307_MERIDIEM);
294  Data += 0x12;
295  }
296  }
297 
298  return DS1307_I2CM_WRITEBYTE(DS1307_REGISTER_HOURS, TRUE) | DS1307_I2CM_WRITEBYTE(Data, TRUE);
299 }

313 const I2C_Error_t DS1307_SetTime(const Time_t* Time)
314 {
315  uint8_t Data[8];
316 
317  Data[0] = 0x00;
318  Data[1] = Dec2BCD(Time->Second);
319  Data[2] = Dec2BCD(Time->Minute);
320  Data[3] = Dec2BCD(Time->Hour);
321  Data[4] = Dec2BCD(Time->DayOfWeek);
322  Data[5] = Dec2BCD(Time->Day);
323  Data[6] = Dec2BCD(Time->Month);
324  Data[7] = Dec2BCD(Time->Year);
325 
326  // Set hour mode
327  if(Time->HourMode == MODE_12_HOUR)
328  {
329  // Check if time is valid and correct it, if necessary. Also set the AM/PM bit
330  if(Time->Hour > 12)
331  {
332  Data[3] = (Time->Hour - 12) | (0x01 << DS1307_MERIDIEM);
333  }
334  else
335  {
336  Data[3] &= ~(0x01 << DS1307_MERIDIEM);
337  }
338 
339  // Set the hour mode bit
340  Data[3] |= (0x01 << DS1307_12_24);
341  }
342 
343  return DS1307_I2CM_WRITEBYTES(sizeof(Data), Data, TRUE);
344 }

346 const I2C_Error_t DS1307_GetTime(Time_t* Time)
347 {
348  uint8_t Buffer[7];
349  I2C_Error_t ErrorCode = I2C_NO_ERROR;
350 
351  ErrorCode = DS1307_I2CM_WRITEBYTE(0x00, FALSE) | DS1307_I2CM_READBYTES(sizeof(Buffer), Buffer, TRUE);
352  if(ErrorCode != I2C_NO_ERROR)
353  {
354  return ErrorCode;
355  }
356 
357  Time->Second = BCD2Dec(Buffer[0]);
358  Time->Minute = BCD2Dec(Buffer[1]);
359 
360  // Check if 12 hour mode is enabled
361  if(Buffer[2] & (0x01 << DS1307_12_24))
362  {
363  Time->HourMode = MODE_12_HOUR;
364  Time->MeridiemMode = (Time->Hour >> DS1307_MERIDIEM) & 0x01;
365 
366  Buffer[2] &= ~((0x01 << 0x05) | (0x01 << 0x06));
367 
368  Time->Hour = BCD2Dec(Buffer[2]);
369 
370  if(Time->Hour > 12)
371  {
372  Time->Hour -= 12;
373  }
374  }
375  else
376  {
377  Time->HourMode = MODE_24_HOUR;
378  Time->Hour = BCD2Dec(Buffer[2]);
379  }
380 
381  Time->DayOfWeek = BCD2Dec(Buffer[3]);
382  Time->Day = BCD2Dec(Buffer[4]);
383  Time->Month = BCD2Dec(Buffer[5]);
384  Time->Year = BCD2Dec(Buffer[6]);
385 
386  return ErrorCode;
387 }
