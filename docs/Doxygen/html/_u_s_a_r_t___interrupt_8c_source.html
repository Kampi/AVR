62 static void __USART_InterruptHandler(const uint8_t Device, const uint8_t Channel, const USART_CallbackType_t Callback)
63 {
64  if(__USART_IsSPI[Device][Channel] == TRUE)
65  {
66  if(Callback == USART_TXC_INTERRUPT)
67  {
68  if(__SPI_Messages[Device][Channel].BytesProcessed < (__SPI_Messages[Device][Channel].Length - 1))
69  {
70  ((USART_t*)(__SPI_Messages[Device][Channel].Device))->DATA = __SPI_Messages[Device][Channel].BufferOut[++__SPI_Messages[Device][Channel].BytesProcessed];
71  }
72  else
73  {
74  USART_SPI_DeselectDevice(__SPI_Messages[Device][Channel].Port, __SPI_Messages[Device][Channel].Pin);
75 
76  if(__USART_SPI_Callbacks[Device][Channel].CompleteInterrupt != NULL)
77  {
78  __USART_SPI_Callbacks[Device][Channel].CompleteInterrupt();
79  }
80 
81  __SPI_Messages[Device][Channel].Status = SPI_MESSAGE_COMPLETE;
82  }
83  }
84  else if(Callback == USART_RXC_INTERRUPT)
85  {
86  __SPI_Messages[Device][Channel].BufferIn[__SPI_Messages[Device][Channel].BytesProcessed] = ((USART_t*)(__SPI_Messages[Device][Channel].Device))->DATA;
87  }
88  }
89  else
90  {
91  if(Callback == USART_DRE_INTERRUPT)
92  {
93  if (__USART_Callbacks[Device][Channel].EmptyCallback)
94  {
95  __USART_Callbacks[Device][Channel].EmptyCallback();
96  }
97 
98  if(!(RingBuffer_IsEmpty(__USART_Messages[Device][Channel].Ptr_TxRingBuffer)))
99  {
100  __USART_Messages[Device][Channel].Device->DATA = RingBuffer_Load(__USART_Messages[Device][Channel].Ptr_TxRingBuffer);
101  }
102  else
103  {
104  __USART_Messages[Device][Channel].Device->CTRLA &= ~0x03;
105  }
106  }
107  else if(Callback == USART_TXC_INTERRUPT)
108  {
109  if (__USART_Callbacks[Device][Channel].TxCallback)
110  {
111  __USART_Callbacks[Device][Channel].TxCallback();
112  }
113  }
114  else if(Callback == USART_RXC_INTERRUPT)
115  {
116  // Echo message when enabled
117  if(__USART_Echo[Device][Channel] == TRUE)
118  {
119  __USART_Messages[Device][Channel].Device->DATA = __USART_Messages[Device][Channel].Device->DATA;
120  }
121 
122  if (__USART_Callbacks[Device][Channel].RxCallback)
123  {
124  __USART_Callbacks[Device][Channel].RxCallback();
125  }
126  }
127  }
128 }
