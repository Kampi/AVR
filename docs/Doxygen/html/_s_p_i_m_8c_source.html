84  GPIO_SetDirection(&PORTD, SPI_MOSI_PIN, GPIO_DIRECTION_OUT);
85  }
86 
87  __SPI_DeviceModes[ID] = SPI_MASTER;
88 
89  Device->CTRL = (Device->CTRL & (~(0x01 << 0x04))) | (SPI_MASTER << 0x04);
90  SPI_SetMode(Device, Config->Mode);
91  SPI_SetDataOrder(Device, Config->DataOrder);
92  SPIM_SetPrescaler(Device, Config->Prescaler);

134 const uint32_t SPIM_GetClock(SPI_t* Device, const uint32_t Clock)
135 {
136  uint8_t Clk2x = ((Device->CTRL & (0x01 << 0x07)) >> 0x07);
137 
138  switch(Device->CTRL & 0x03)
139  {
140  case(0x01):
141  {
142  return (Clock >> (0x04 >> Clk2x));
143  }
144  case(0x02):
145  {
146  return (Clock >> (0x06 >> Clk2x));
147  }
148  case(0x03):
149  {
150  return (Clock >> (0x07 >> Clk2x));
151  }
152  default:
153  {
154  return (Clock >> (0x02 >> Clk2x));
155  }
156  }
