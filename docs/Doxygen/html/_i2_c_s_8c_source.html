41 void I2CS_Init(I2CS_Config_t* Config)
42 {
43  uint8_t ID = 0x00;
44 
45  if(Config->Device == &TWIC)
46  {
47  ID = TWIC_ID;
48  }
49  else if(Config->Device == &TWIE)
50  {
51  ID = TWIE_ID;
52  }
53 
54  I2C_PowerEnable(Config->Device);
55 
56  I2C_SlaveEnable(Config->Device);
57  I2CS_SetAddress(Config->Device, Config->Address);
58  I2CS_SetMask(Config->Device, Config->Mask);
59  I2CS_SwitchSmartMode(Config->Device, Config->EnableSmartMode);
60  I2CS_SwitchPromiscuousMode(Config->Device, Config->EnablePromiscuousMode);
61 
62  __I2CS_Buffer[ID].Buffer = Config->Buffer;
63  __I2CS_Buffer[ID].Device = Config->Device;
64 
65  Config->Device->SLAVE.CTRLA |= TWI_SLAVE_DIEN_bm | TWI_SLAVE_APIEN_bm | TWI_SLAVE_PIEN_bm;
66  Config->Device->SLAVE.CTRLA = (Config->Device->SLAVE.CTRLA & (~(0x03 << 0x06))) | (Config->InterruptLevel << 0x06);
67 }
