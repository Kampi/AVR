49 static inline void Timer1_InterruptHandler(const uint8_t Device, const Timer_CallbackType_t Callback)
50 {
51  DisableGlobalInterrupts();
52 
53  if(Callback == TIMER_OVERFLOW_INTERRUPT)
54  {
55  if(Timer1_Callbacks[Device].OverflowCallback != NULL)
56  {
57  Timer1_Callbacks[Device].OverflowCallback();
58  }
59  }
60  else if(Callback == TIMER_CAPTURE_INTERRUPT)
61  {
62  if(Timer1_Callbacks[Device].CaptureCallback != NULL)
63  {
64  Timer1_Callbacks[Device].CaptureCallback();
65  }
66  }
67  else if(Callback == TIMER_COMPMATCH_A_INTERRUPT)
68  {
69  if(Timer1_Callbacks[Device].CompareACallback != NULL)
70  {
71  Timer1_Callbacks[Device].CompareACallback();
72  }
73  }

75  {
76  if(Timer1_Callbacks[Device].CompareBCallback != NULL)
77  {
78  Timer1_Callbacks[Device].CompareBCallback();

83 }
