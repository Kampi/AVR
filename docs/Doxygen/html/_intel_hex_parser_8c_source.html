69 static uint16_t Hex2Num(const uint8_t Length)
70 {
71  uint16_t Temp = 0x00;
72 
73  for(uint8_t i = 0x00; i < Length; i++)
74  {
75  uint8_t c = __ParserBuffer[__Index++];
76 
77  if(c >= '0' && c <= '9')
78  {
79  c -= '0';
80  }
81  else if(c >= 'A' && c <= 'F')
82  {
83  c -= 'A' - 10;
84  }
85  else if(c >= 'a' && c <= 'f')
86  {
87  c -= 'a' - 10;
88  }
89 
90  Temp = (Temp << 0x04) + c;
91  }
92 
93  return Temp;
94 }
