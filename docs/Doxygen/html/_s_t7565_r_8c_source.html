116 {

142 void Display_Init(SPIM_Config_t* Config)
143 {
144  const uint8_t Init[] = {
145  ST7565R_CMD_ADC_NORMAL,
146  ST7565R_CMD_DISPLAY_NORMAL,
147  ST7565R_CMD_REVERSE_SCAN_DIRECTION,
148  ST7565R_CMD_LCD_BIAS_1_DIV_6_DUTY33,
149  ST7565R_CMD_POWER_CTRL_ALL_ON,
150  ST7565R_CMD_BOOSTER_RATIO,
151  ST7565R_CMD_BOOSTER_RATIO_2X_3X_4X,
152  ST7565R_CMD_VOLTAGE_RESISTOR_RATIO_1,
153  ST7565R_CMD_ELECTRONIC_VOLUME_MODE_SET,
154  ST7565R_CMD_ELECTRONIC_VOLUME(DISPLAY_CONTRAST_MIN),
155  ST7565R_CMD_DISPLAY_ON,
156  };
157 
158  // Set display back light as output in low state
159  GPIO_SetDirection(GET_PERIPHERAL(ST7565R_BACKLIGHT), GET_INDEX(ST7565R_BACKLIGHT), GPIO_DIRECTION_OUT);
160  GPIO_Clear(GET_PERIPHERAL(ST7565R_BACKLIGHT), GET_INDEX(ST7565R_BACKLIGHT));
161 
162  // Set display reset as output in high state
163  GPIO_SetDirection(GET_PERIPHERAL(ST7565R_RESET), GET_INDEX(ST7565R_RESET), GPIO_DIRECTION_OUT);
164  GPIO_Set(GET_PERIPHERAL(ST7565R_RESET), GET_INDEX(ST7565R_RESET));
165 
166  // Set register select as output in high state
167  GPIO_SetDirection(GET_PERIPHERAL(ST7565R_REGISTER_SELECT), GET_INDEX(ST7565R_REGISTER_SELECT), GPIO_DIRECTION_OUT);
168  GPIO_Set(GET_PERIPHERAL(ST7565R_REGISTER_SELECT), GET_INDEX(ST7565R_REGISTER_SELECT));
169 
170  // Set display SS as output in high state
171  GPIO_SetDirection(GET_PERIPHERAL(ST7565R_SS), GET_INDEX(ST7565R_SS), GPIO_DIRECTION_OUT);
172  GPIO_Set(GET_PERIPHERAL(ST7565R_SS), GET_INDEX(ST7565R_SS));
173 
174  // Reset the display
175  ST7565R_HardwareReset();
176 
177  // Initialize SPI-Interface
178  if(Config != NULL)
179  {
180  ST7565R_SPIM_INIT(Config);
181  }
182 
183  // Set A0 to low state for "Command"
184  GPIO_Clear(GET_PERIPHERAL(ST7565R_REGISTER_SELECT), GET_INDEX(ST7565R_REGISTER_SELECT));
185 
186  ST7565R_WriteCommandBytes(Init, sizeof(Init));
187 
188  // Clear Display
189  ST7565R_Clear();
190 }
