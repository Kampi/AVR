164 static const I2C_Error_t PCA9685_SwitchBit(const uint8_t Register, const uint8_t Mask, const Bool_t Set)
165 {
166  uint8_t Data[2] = {Register, 0x00};
167  I2C_Error_t ErrorCode = I2C_NO_ERROR;
168 
169  ErrorCode = PCA9685_I2CM_WRITEBYTE(Data[0], FALSE) | PCA9685_I2CM_READBYTE(&Data[1], TRUE);
170  if(ErrorCode != I2C_NO_ERROR)
171  {
172  return ErrorCode;
173  }
174 
175  if(Set == TRUE)
176  {
177  Data[1] |= Mask;
178  }
179  else
180  {
181  Data[1] &= ~Mask;
182  }
183 
184  return PCA9685_I2CM_WRITEBYTES(sizeof(Data), Data, TRUE);
185 }

242 const I2C_Error_t PCA9685_SetOutputEnable(const PCA9685_OutputEnable_t Mode)
243 {
244  uint8_t Data[2] = {PCA9685_REGISTER_MODE2, 0x00};
245  I2C_Error_t ErrorCode = I2C_NO_ERROR;
246 
247  ErrorCode = PCA9685_I2CM_WRITEBYTE(Data[0], FALSE) | PCA9685_I2CM_READBYTE(&Data[1], TRUE);
248  if(ErrorCode != I2C_NO_ERROR)
249  {
250  return ErrorCode;
251  }
252 
253  if(Mode == PCA9685_OUTNE_MODE_0)
254  {
255  Data[1] &= ~((0x01 << PCA9685_OUTNE1) | (0x01 << PCA9685_OUTNE0));
256  }
257  else if(Mode == PCA9685_OUTNE_MODE_1)
258  {
259  Data[1] &= ~((0x01 << PCA9685_OUTNE1) | (0x01 << PCA9685_OUTNE0));
260  Data[1] |= (0x01 << PCA9685_OUTNE0);
261  }
262  else
263  {
264  Data[1] &= ~((0x01 << PCA9685_OUTNE1) | (0x01 << PCA9685_OUTNE0));
265  Data[1] |= (0x01 << PCA9685_OUTNE1) | (0x01 << PCA9685_OUTNE0);
266  }
267 
268  return PCA9685_I2CM_WRITEBYTES(sizeof(Data), Data, TRUE);
269 }

349 const I2C_Error_t PCA9685_SetChannel(const PCA9685_Channel_t Channel, const uint16_t On, const uint16_t Off)
350 {
351  /*
352  Message packet
353  0 = Start address (LEDn_ON_L)
354  1 = ON_L value
355  2 = ON_H value
356  3 = OFF_L value
357  4 = OFF_H value
358  */
359  uint8_t Data[5];
360 
361  // Get the PWM channel
362  if(Channel == PCA9685_CHANNEL_ALL)
363  {
364  Data[0] = PCA9685_REGISTER_ALL_LED_ON_L;
365  }
366  else
367  {
368  Data[0] = PCA9685_REGISTER_LED0_ON_L + (Channel << 0x02);
369  }
370 
371  // Save the On/Off time
372  Data[1] = On & 0xFF;
373  Data[2] = On >> 0x08;
374  Data[3] = Off & 0xFF;
375  Data[4] = Off >> 0x08;
376 
377  return PCA9685_I2CM_WRITEBYTES(sizeof(Data), Data, TRUE);
378 }

381 {
382  uint16_t On = 0x00;
383  uint16_t Off = 0x00;
384 
385  if(Duty > 100.0)
386  {
387  return I2C_INVALID_PARAM;
388  }
389 
390  Off = (PCA9685_TIME_RANGE / 100.0) * Duty;
391 
392  return PCA9685_SetChannel(Channel, On, Off);
