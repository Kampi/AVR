102  Port = GPIO_PORT_F;
103  }

105  {
106  Port = GPIO_PORT_R;
107  }
108 
109  if(Config->Channel & GPIO_INTERRUPT_0)
110  {
111  Config->Port->INTCTRL |= (Config->Port->INTCTRL & ~PORT_INT0LVL_gm) | Config->InterruptLevel;
112  Config->Port->INT0MASK |= (Config->Port->INT0MASK & (~(0x01 << Config->Pin))) | (0x01 << Config->Pin);
113  GPIO_Callbacks[Port].Int0 = Config->Callback;
114  }
115 
116  if(Config->Channel & GPIO_INTERRUPT_1)
117  {
118  Config->Port->INTCTRL |= (Config->Port->INTCTRL & ~PORT_INT1LVL_gm) | (Config->InterruptLevel << 0x03);
119  Config->Port->INT1MASK |= (Config->Port->INT1MASK & (~(0x01 << Config->Pin))) | (0x01 << Config->Pin);
120  GPIO_Callbacks[Port].Int1 = Config->Callback;
121  }

123  GPIO_SetConfig(Config->Port, Config->Pin, Config->Sense | Config->PullConfig);
124 }

138 void GPIO_ChangeInterruptLevel(PORT_t* Port, const uint8_t Pin, const GPIO_InterruptChannel_t Channel, const Interrupt_Level_t InterruptLevel)
139 {
140  if(Channel && GPIO_INTERRUPT_0)
141  {
142  Port->INTCTRL = (Port->INTCTRL & ~PORT_INT0LVL_gm) | InterruptLevel;
143  Port->INT0MASK = (Port->INT0MASK & (~(0x01 << Pin))) | (0x01 << Pin);
144  }
145 
146  if(Channel && GPIO_INTERRUPT_1)
147  {
148  Port->INTCTRL |= (Port->INTCTRL & ~PORT_INT1LVL_gm) | (InterruptLevel << 0x03);
149  Port->INT1MASK |= (Port->INT1MASK & (~(0x01 << Pin))) | (0x01 << Pin);
150  }
