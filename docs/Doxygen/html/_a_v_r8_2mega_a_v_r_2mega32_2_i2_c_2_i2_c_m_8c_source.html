38 static I2C_Status_t I2CM_SendStart(void)
39 {
40  TWCR = (0x01 << TWINT) | (0x01 << TWEN) | (0x01 << TWSTA);
41 
42  while(!(TWCR & (0x01 << TWINT)));

103 void I2CM_SetBaudrate(const uint32_t Bitrate, const uint8_t Prescaler, const uint32_t Clock)
104 {
105  uint8_t Exp = 0x01;
106 
107  for(uint8_t i = 0x00; i < Prescaler; i++)
108  {
109  Exp = Exp << 0x02;
110  }
111 
112  TWBR = ((Clock / Bitrate) - 0x10) / (Exp << 0x01);

149 I2C_Error_t I2CM_WriteByte(const uint8_t Address, const uint8_t Data, const Bool_t Stop)
150 {
151  I2C_Status_t Status = 0x00;
152 
153  Status = I2CM_SendStart();
154  if((Status != I2C_START_SEND) & (Status != I2C_REP_START_SEND))
155  {
156  return I2C_START_ERROR;
157  }
158 
159  if(I2CM_Send(I2C_WRITE(Address)) != I2C_ADDR_ACK_W_RECEIVED)
160  {
161  return I2C_WRITE_ADDR_ERROR;
162  }
163 
164  if(I2CM_Send(Data) != I2C_DATA_ACK_W_RECEIVED)
165  {
166  return I2C_BUS_ERROR;
167  }
168 
169  if(Stop == TRUE)
170  {
171  I2CM_SendStop();
172  }
173 
174  return I2C_NO_ERROR;
175 }
