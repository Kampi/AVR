110 void DAC_EnableChannel(DAC_t* Device, const DAC_Channel_t Channel)
111 {
112  if(Channel & DAC_CHANNEL_0)
113  {
114  Device->CTRLA |= (0x01 << 0x02);
115  }
116  else if(Channel & DAC_CHANNEL_1)
117  {
118  Device->CTRLA |= (0x01 << 0x03);
119  }
120  else if(Channel & DAC_CHANNEL_INT)
121  {
122  Device->CTRLA &= ~(0x01 << 0x03);
123  Device->CTRLA |= (0x01 << 0x04) | (0x01 << 0x02);
124  }
125 }

185 void DAC_WriteChannel(DAC_t* Device, const DAC_Channel_t Channel, const uint16_t Value)
186 {
187  if((Channel & DAC_CHANNEL_0) || (Channel & DAC_CHANNEL_INT))
188  {
189  while(!(Device->STATUS & DAC_CH0DRE_bm));
190  Device->CH0DATA = Value;
191  }
192 
193  if(Channel & DAC_CHANNEL_1)
194  {
195  while(!(Device->STATUS & DAC_CH1DRE_bm));
196  Device->CH1DATA = Value;
197  }
198 }
