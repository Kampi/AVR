152 static void HD44780_WriteByte(const uint8_t Data)
153 {
154  HD44780_ClearData();
155 
156  #if(defined(HD44780_USE_4BIT))
157  // Send the upper nibble
158  GPIO_SetPort(GET_PERIPHERAL(HD44780_DATA), (Data & 0xF0) >> (0x04 - GET_INDEX(HD44780_DATA)));
159 
160  // Strobe the ENABLE pin
161  HD44780_Strobe();
162 
163  // Send the lower nibble
164  HD44780_ClearData();
165  GPIO_SetPort(GET_PERIPHERAL(HD44780_DATA), (Data & 0x0F) << GET_INDEX(HD44780_DATA));
166 
167  // Strobe the ENABLE pin
168  HD44780_Strobe();
169  #elif(defined(HD44780_USE_8BIT))
170  GPIO_SetPort(GET_PERIPHERAL(HD44780_DATA), Data);
171 
172  // Strobe the ENABLE pin
173  HD44780_Strobe();
174  #endif
175 }

179 static void HD44780_Reset(void)
180 {
181  // Wait until the display is ready
182  _delay_ms(15);
183 
184  GPIO_Clear(GET_PERIPHERAL(HD44780_ENABLE), GET_INDEX(HD44780_ENABLE));
185  GPIO_Clear(GET_PERIPHERAL(HD44780_RS), GET_INDEX(HD44780_RS));
186  GPIO_Clear(GET_PERIPHERAL(HD44780_RW), GET_INDEX(HD44780_RW));
187 
188  HD44780_ClearData();
189 
190  // You have to write 0x30 three times to reset the display.
191  // NOTE: The display use 8 bit mode after the reset!
192  __HD44780_Func |= HD44780_FUNC_8BIT;
193  #if(defined(HD44780_USE_4BIT))
194  GPIO_SetPort(GET_PERIPHERAL(HD44780_DATA), 0x03 << GET_INDEX(HD44780_DATA));
195  #elif(defined(HD44780_USE_8BIT))
196  GPIO_SetPort(GET_PERIPHERAL(HD44780_DATA), 0x30);
197  #endif
198 
199  for(uint8_t i = 0x00; i < 0x03; i++)
200  {
201  HD44780_Strobe();
202  _delay_ms(5);
203  }
204 
205  #if(defined(HD44780_USE_4BIT))
206  HD44780_ClearData();
207  // Switch the display into 4 bit mode
208  __HD44780_Func &= ~HD44780_FUNC_8BIT;
209  GPIO_SetPort(GET_PERIPHERAL(HD44780_DATA), 0x02 << GET_INDEX(HD44780_DATA));
210  #endif
211 
212  HD44780_Strobe();
213 
214  _delay_us(5);
215 }

221  static Bool_t HD44780_IsBusy(void)
222  {
223  volatile uint8_t Status = 0x00;
224 
225  // Switch DB7 to input
226  #if(defined(HD44780_USE_8BIT))
227  GPIO_SetDirection(GET_PERIPHERAL(HD44780_DATA), GET_INDEX(HD44780_DATA) + 0x07, GPIO_DIRECTION_IN);
228  #elif(defined(HD44780_USE_4BIT))
229  GPIO_SetDirection(GET_PERIPHERAL(HD44780_DATA), GET_INDEX(HD44780_DATA) + 0x03, GPIO_DIRECTION_IN);
230  #endif
231 
232  GPIO_Clear(GET_PERIPHERAL(HD44780_RS), GET_INDEX(HD44780_RS));
233  GPIO_Set(GET_PERIPHERAL(HD44780_RW), GET_INDEX(HD44780_RW));
234  _delay_us(1);
235 
236  // Read the BUSY bit
237  GPIO_Set(GET_PERIPHERAL(HD44780_ENABLE), GET_INDEX(HD44780_ENABLE));
238  _delay_us(1);
239  #if(defined(HD44780_USE_8BIT))
240  Status = GPIO_Read(GET_PERIPHERAL(HD44780_DATA), GET_INDEX(HD44780_DATA) + 0x07) >> 0x07;
241  #elif(defined(HD44780_USE_4BIT))
242  Status = GPIO_Read(GET_PERIPHERAL(HD44780_DATA), GET_INDEX(HD44780_DATA) + 0x03) >> 0x03;
243  #endif
244  GPIO_Clear(GET_PERIPHERAL(HD44780_ENABLE), GET_INDEX(HD44780_ENABLE));
245 
246  #if(defined(HD44780_USE_4BIT))
247  // Read the lower nibble
248  GPIO_Set(GET_PERIPHERAL(HD44780_ENABLE), GET_INDEX(HD44780_ENABLE));
249  _delay_us(1);
250  GPIO_Clear(GET_PERIPHERAL(HD44780_ENABLE), GET_INDEX(HD44780_ENABLE));
251  #endif
252 
253  GPIO_Set(GET_PERIPHERAL(HD44780_RS), GET_INDEX(HD44780_RS));
254  GPIO_Clear(GET_PERIPHERAL(HD44780_RW), GET_INDEX(HD44780_RW));
255 
256  // Switch DB7 back to output
257  #if(defined(HD44780_USE_8BIT))
258  GPIO_SetDirection(GET_PERIPHERAL(HD44780_DATA), GET_INDEX(HD44780_DATA) + 0x07, GPIO_DIRECTION_OUT);
259  #elif(defined(HD44780_USE_4BIT))
260  GPIO_SetDirection(GET_PERIPHERAL(HD44780_DATA), GET_INDEX(HD44780_DATA) + 0x03, GPIO_DIRECTION_OUT);
261  #endif
262

264  }

299 void HD44780_Init(void)
300 {
301  __HD44780_Entry = HD44780_CMD_ENTRY;
302  __HD44780_Display = HD44780_CMD_DISPLAY;
303  __HD44780_Cursor = HD44780_CMD_CURSOR;
304  __HD44780_Func = HD44780_CMD_FUNC;
305 
306  GPIO_SetDirection(GET_PERIPHERAL(HD44780_BACKLIGHT), GET_INDEX(HD44780_BACKLIGHT), GPIO_DIRECTION_OUT);
307  GPIO_SetDirection(GET_PERIPHERAL(HD44780_ENABLE), GET_INDEX(HD44780_ENABLE), GPIO_DIRECTION_OUT);
308  GPIO_SetDirection(GET_PERIPHERAL(HD44780_RS), GET_INDEX(HD44780_RS), GPIO_DIRECTION_OUT);
309  GPIO_SetDirection(GET_PERIPHERAL(HD44780_RW), GET_INDEX(HD44780_RW), GPIO_DIRECTION_OUT);
310 
311  // Configure the data pins as output
312  #if(defined(HD44780_USE_8BIT))
313  GPIO_SetPortDirection(GET_PERIPHERAL(HD44780_DATA), 0xFF, GPIO_DIRECTION_OUT);
314  #else
315  GPIO_SetPortDirection(GET_PERIPHERAL(HD44780_DATA), 0x0F << GET_INDEX(HD44780_DATA), GPIO_DIRECTION_OUT);
316  #endif
317 
318  HD44780_ClearData();
319  GPIO_Clear(GET_PERIPHERAL(HD44780_ENABLE), GET_INDEX(HD44780_ENABLE));
320  GPIO_Clear(GET_PERIPHERAL(HD44780_RS), GET_INDEX(HD44780_RS));
321  GPIO_Clear(GET_PERIPHERAL(HD44780_RW), GET_INDEX(HD44780_RW));
322 
323  // Reset the display
324  HD44780_Reset();
325 
326  // Set the display lines
327  #if(HD44780_LINES > 1)
328  __HD44780_Func |= HD44780_FUNC_2LINE;
329  #else
330  __HD44780_Func &= ~HD44780_FUNC_1LINE;
331  #endif
332  HD44780_SendCommand(__HD44780_Func);
333 
334  // Set the entry options
335  __HD44780_Entry |= HD44780_ENTRY_INCR | HD44780_ENTRY_SHIFT_OFF;
336  HD44780_SendCommand(__HD44780_Entry);
337 
338  // Set the cursor options
339  __HD44780_Cursor |= HD44780_CURSOR_CHARCTER_MODE | HD44780_CURSOR_POWER_ON | HD44780_CURSOR_MODE_MASK;
340  HD44780_SendCommand(__HD44780_Cursor);
341 
342  // Set the display options
343  __HD44780_Display |= HD44780_DISPLAY_ON;
344  HD44780_SendCommand(__HD44780_Display);
345 
346  // Move the cursor to the home position and clear the display
347  HD44780_ClearHome();
348 }
