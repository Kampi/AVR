74 void Bootloader_Init(void)
75 {
76  // Remove the old application
77  NVM_EraseApplication();
78 
79  // Enable the default clock
80  asm volatile( "movw r30, %0" "\n\t"
81  "ldi r16, %2" "\n\t"
82  "out %3, r16" "\n\t"
83  "st Z, %1" "\n\t"
84  :: "r" (&CLK.CTRL),
85  "r" (0x00),
86  "M" (CCP_IOREG_gc),
87  "i" (&CCP)
88  : "r16", "r30"
89  );
90 
91  // Enable the USART interface
92  ((PORT_t*)(&PORT_NAME(BOOTLOADER_INTERFACE)))->DIRSET = (0x01 << BOOTLOADER_TX);
93  ((PORT_t*)(&PORT_NAME(BOOTLOADER_INTERFACE)))->DIRCLR = (0x01 << BOOTLOADER_RX);
94  ((USART_t*)(&USART_NAME(BOOTLOADER_INTERFACE)))->BAUDCTRLA = BOOTLOADER_BRREG_VALUE;
95  ((USART_t*)(&USART_NAME(BOOTLOADER_INTERFACE)))->BAUDCTRLB = (BOOTLOADER_SCALE_VALUE << USART_BSCALE_gp) & USART_BSCALE_gm;
96  ((USART_t*)(&USART_NAME(BOOTLOADER_INTERFACE)))->CTRLB = USART_RXEN_bm | USART_TXEN_bm;
97 
98  // Initialize the hex file parser
99  Parser_Init();
100 
101  // Disable the targets flow control
102  Bootloader_PutChar(XON);
103 }

105 Bool_t Bootloader_Enter(void)
106 {
107  Bootloader_PutString("Enter bootloader...\n\r");
108 
109  uint16_t Word = 0x00;
110  uint16_t Page = 0x00;
111 
112  do
113  {
114  Parser_State_t LineRead = Parser_GetLine(Bootloader_GetChar());
115  if(LineRead == PARSER_STATE_SUCCESSFULL)
116  {
117  Bootloader_PutChar(XOFF);
118 
119  Parser_State_t State = IntelParser_ParseLine(&__Line);
120  if(State == PARSER_STATE_SUCCESSFULL)
121  {
122  if(__Line.Type == PARSER_TYPE_DATA)
123  {
124  for(uint8_t i = 0x00; i < __Line.Bytes; i = i + 0x02)
125  {
126  uint16_t CodeWord = (__LineBuffer[i + 1] << 0x08) | __LineBuffer[i];
127  NVM_LoadFlashBuffer((__Line.Address + i) >> 0x01, CodeWord);
128  }
129 
130  Word += __Line.Bytes;
131 
132  if(Word == APP_SECTION_PAGE_SIZE)
133  {
134  Word = 0x00;
135  NVM_FlushFlash(Page++);
136  }
137  }
138  }
139  else if(State == PARSER_STATE_ERROR)
140  {
141  // Error handling
142 
143  return FALSE;
144  }
145 
146  Bootloader_PutChar(XON);
147  }
148  } while(__Line.Type != PARSER_TYPE_EOF);
149 
150  NVM_FlushFlash(Page);
151 
152  return TRUE;
153 }
