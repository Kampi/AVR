115 void ADC_Channel_InstallCallback(ADC_InterruptConfig_t* Config)
116 {
117  uint8_t Device = ADCA_ID;
118  uint8_t Channel = 0x00;
119 
120  #if(defined ADCB)
121  if(Config->Device == &ADCB)
122  {
123  Device = ADCB_ID;
124  }
125  #endif
126 
127  if(Config->Channel == &Config->Device->CH0)
128  {
129  Channel = 0x00;
130  }
131  #if(MCU_NAME == MCU_NAME_ATXMEGA256A3BU)
132  else if(Config->Channel == &Config->Device->CH1)
133  {
134  Channel = 0x01;
135  }
136  else if(Config->Channel == &Config->Device->CH2)
137  {
138  Channel = 0x02;
139  }
140  else if(Config->Channel == &Config->Device->CH3)
141  {
142  Channel = 0x03;
143  }
144  #endif
145 
146  if(Config->Source & ADC_CONVERSION_COMPLETE_INTERRUPT)
147  {
148  ADC_Callbacks[Device][Channel].ConversionComplete = Config->Callback;
149  }
150 
151  if(Config->Source & ADC_THRESH_BELOW_INTERRUPT)
152  {
153  ADC_Callbacks[Device][Channel].ThreshBelow = Config->Callback;
154  }
155 
156  if(Config->Source & ADC_THRESH_ABOVE_INTERRUPT)
157  {
158  ADC_Callbacks[Device][Channel].ThreshAbove = Config->Callback;
159  }
160 
161  Config->Channel->INTCTRL |= (Config->Channel->INTCTRL & (~Config->InterruptLevel)) | Config->InterruptLevel;
162  Config->Channel->INTCTRL |= (Config->Channel->INTCTRL & (~(Config->Source << 0x02))) | (Config->Source << 0x02);
163 }

170 void ADC_SetupInput(ADC_ChannelConfig_t* Config)
171 {
172  if(Config->Device == &ADCA)
173  {
174  GPIO_SetDirection(&PORTA, Config->Input, GPIO_DIRECTION_IN);
175  GPIO_DisableBuffer(&PORTA, Config->Input);
176  }
177 
178  #if(defined ADCB)
179  if(Config->Device == &ADCB)
180  {
181  GPIO_SetDirection(&PORTB, Config->Input, GPIO_DIRECTION_IN);
182  GPIO_DisableBuffer(&PORTA, Config->Input);
183  }
184  #endif
185 
186  Config->Channel->CTRL |= (Config->Channel->CTRL & (~Config->Mode)) | Config->Mode;
187  Config->Channel->CTRL |= (Config->Channel->CTRL & (~(Config->Gain << 0x02))) | (Config->Gain << 0x02);
188  Config->Channel->MUXCTRL |= (Config->Channel->MUXCTRL & (~(Config->Input << 0x03))) | (Config->Input << 0x03);
189 }

191 void ADC_Calibrate(ADC_t* Device)
192 {
193  uint8_t OffsetL = 0x00;
194  uint8_t OffsetH = 0x00;
195 
196  if(Device == &ADCA)
197  {
198  OffsetL = offsetof(NVM_PROD_SIGNATURES_t, ADCACAL0);
199  OffsetH = offsetof(NVM_PROD_SIGNATURES_t, ADCACAL1);
200  }
201  else
202  {
203  #if(defined ADCB)
204  OffsetL = offsetof(NVM_PROD_SIGNATURES_t, ADCBCAL0);
205  OffsetH = offsetof(NVM_PROD_SIGNATURES_t, ADCBCAL1);
206  #endif
207  }
208 
209  Device->CALL = NVM_ReadCalibrationByte(OffsetL);
210  Device->CALH = NVM_ReadCalibrationByte(OffsetH);
