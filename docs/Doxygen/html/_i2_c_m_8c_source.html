85 static uint8_t I2CM_ReadData(TWI_t* Device, const Bool_t NACK)
86 {
87  uint8_t Data = 0x00;
88 
89  while(!(Device->MASTER.STATUS & TWI_MASTER_RIF_bm));
90  Data = Device->MASTER.DATA;
91 
92  if(!NACK)
93  {
94  Device->MASTER.CTRLC = TWI_MASTER_CMD_RECVTRANS_gc;
95  }
96 
97  return Data;
98 }

134 I2C_Error_t I2CM_WriteByte(TWI_t* Device, const uint8_t Address, const uint8_t Data, const Bool_t Stop)
135 {
136  I2CM_SendAddress(Device, I2C_WRITE(Address));
137  if(I2CM_ReadStatus(Device) & TWI_MASTER_BUSERR_bm)
138  {
139  return I2C_WRITE_ADDR_ERROR;
140  }
141 
142  I2CM_SendData(Device, Data);
143  if(I2CM_ReadStatus(Device) & (TWI_MASTER_BUSERR_bm | TWI_MASTER_RXACK_bm | TWI_MASTER_ARBLOST_bm))
144  {
145  I2CM_SendStop(Device, FALSE);
146 
147  return I2C_BUS_ERROR;
148  }
149 
150  if(Stop == TRUE)
151  {
152  I2CM_SendStop(Device, FALSE);
153  }
154 
155  return I2C_NO_ERROR;
156 }

232 void I2CM_SetBaudrate(TWI_t* Device, const uint32_t Baudrate, const uint32_t Clock)
233 {
234  uint32_t BaudTemp = Baudrate;
235 
236  if((BaudTemp < 100000) || (BaudTemp > 400000))
237  {
238  BaudTemp = 100000;
239  }
240 
241  BaudTemp = (Clock / (BaudTemp << 0x01)) - 5;
242 
243  Device->MASTER.BAUD = BaudTemp;
244 }
