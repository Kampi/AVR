58 static OneWire_Error_t DS18B20_SearchDevices(uint8_t* Found, uint8_t Search, OneWire_ROM_t* ROM, const Bool_t isAlarm)
59 {
60  uint8_t DevicesFound = 0x00;
61 
62  if(Found == NULL)
63  {
64  return ONEWIRE_PARAMETER_ERROR;
65  }
66 
67  OneWire_Error_t ErrorCode = OneWire_StartSearch(ROM, isAlarm);
68  if(ErrorCode == ONEWIRE_NO_ERROR)
69  {
70  if(ROM->FamilyCode == DS18B20_ID)
71  {
72  DevicesFound++;
73  ROM++;
74  }
75 
76  // Search additional devices
77  while((!OneWire_IsLast()) && (DevicesFound < Search))
78  {
79  ErrorCode = OneWire_SearchNext(ROM);
80  if(ErrorCode != ONEWIRE_NO_ERROR)
81  {
82  OneWire_StopSearch();
83  break;
84  }
85 
86  if(ROM->FamilyCode == DS18B20_ID)
87  {
88  DevicesFound++;
89  ROM++;
90  }
91  }
92  }
93 
94  *Found = DevicesFound;
95 
96  return ErrorCode;
97 }

268 OneWire_Error_t DS18B20_ReadScratchpad(const OneWire_ROM_t* ROM, const uint8_t Length, DS18B20_Scratchpad_t* Data)
269 {
270  uint8_t* Data_Temp = (uint8_t*)Data;
271 
272  if(Data_Temp == NULL)
273  {
274  return ONEWIRE_PARAMETER_ERROR;
275  }
276 
277  OneWire_Error_t ErrorCode = OneWire_SelectDevice(ROM);
278  if(ErrorCode != ONEWIRE_NO_ERROR)
279  {
280  return ErrorCode;
281  }
282 
283  OneWire_WriteByte(DS18B20_READ_SCRATCHPAD);
284 
285  // Read the scratchpad
286  for(uint8_t i = 0x00; i < Length; i++)
287  {
288  // Data is transmitted LSB first.
289  *(Data_Temp + i) = OneWire_ReadByte();
290  }
291 
292  // Abort transmission when all data are read
293  if(Length < DS18B20_SCRATCHPAD_SIZE)
294  {
295  return OneWire_Reset();
296  }
297 
298  return ONEWIRE_NO_ERROR;
299 }

301 OneWire_Error_t DS18B20_Measure(const OneWire_ROM_t* ROM, const DS18B20_Resolution_t Resolution, double* Temperature)
302 {
303  uint8_t Temp[2];
304  uint8_t Configuration[5];
305 
306  if(Temperature == NULL)
307  {
308  return ONEWIRE_PARAMETER_ERROR;
309  }
310 
311  // Read the scratchpad to get TH, TL and the configuration value
312  OneWire_Error_t ErrorCode = DS18B20_ReadScratchpad(ROM, sizeof(Configuration), (DS18B20_Scratchpad_t*)Configuration);
313  if(ErrorCode != ONEWIRE_NO_ERROR)
314  {
315  return ErrorCode;
316  }
317 
318  // Set the resolution
319  Configuration[4] &= ~0x60;
320  Configuration[4] |= (Resolution & 0x03) << 0x05;
321 
322  // Write the configuration to the device
323  ErrorCode = DS18B20_WriteScratchpad(ROM, Configuration[2], Configuration[3], Configuration[4]);
324  if(ErrorCode != ONEWIRE_NO_ERROR)
325  {
326  return ErrorCode;
327  }
328 
329  #if(defined(ONEWIRE_USE_EXT_PULL))
330  OneWire_PowerState_t UseParasite;
331 
332  // Check the power state of the device
333  ErrorCode = DS18B20_CheckSupply(ROM, &UseParasite);
334  if(ErrorCode != ONEWIRE_NO_ERROR)
335  {
336  return ErrorCode;
337  }
338  #endif
339 
340  // Start a new temperature measurement
341  ErrorCode = OneWire_SelectDevice(ROM);
342  if(ErrorCode != ONEWIRE_NO_ERROR)
343  {
344  return ErrorCode;
345  }
346  OneWire_WriteByte(DS18B20_CONVERT_T);
347 
348  // Wait for the end of conversion
349  #if(defined(ONEWIRE_USE_EXT_PULL))
350  // Only use the delay when parasite power is used. Otherwise poll the device.
351  if(UseParasite == ONEWIRE_POWER_PARSITIC)
352  {
353  // Enable the external pull up
354  #if(defined(ONEWIRE_EXT_PULL_ACTIVE_LOW))
355  GPIO_Clear(GET_PERIPHERAL(ONEWIRE_EXT_PULL), GET_INDEX(ONEWIRE_EXT_PULL));
356  #else
357  GPIO_Set(GET_PERIPHERAL(ONEWIRE_EXT_PULL), GET_INDEX(ONEWIRE_EXT_PULL));
358  #endif
359 
360  if(Resolution == DS18B20_RESOLUTION_9) _delay_ms(100);
361  else if(Resolution == DS18B20_RESOLUTION_9) _delay_ms(200);
362  else if(Resolution == DS18B20_RESOLUTION_9) _delay_ms(400);
363  else _delay_ms(8000);
364 
365  // Disable the external pull up
366  #if(defined(ONEWIRE_EXT_PULL_ACTIVE_LOW))
367  GPIO_Set(GET_PERIPHERAL(ONEWIRE_EXT_PULL), GET_INDEX(ONEWIRE_EXT_PULL));
368  #else
369  GPIO_Clear(GET_PERIPHERAL(ONEWIRE_EXT_PULL), GET_INDEX(ONEWIRE_EXT_PULL));
370  #endif
371  }
372  else
373  {
374  uint8_t Status = 0x00;
375 
376  do
377  {
378  Status = OneWire_ReadBit();
379  }while(Status == 0x00);
380  }
381  #else
382  uint8_t Status = 0x00;
383 
384  do
385  {
386  Status = OneWire_ReadBit();
387  }while(Status == 0x00);
388  #endif
389 
390  // Get the temperature data
391  ErrorCode = DS18B20_ReadScratchpad(ROM, sizeof(Temp), (DS18B20_Scratchpad_t*)Temp);
392  if(ErrorCode != ONEWIRE_NO_ERROR)
393  {
394  return ErrorCode;
395  }
396 
397  #if(defined(MCU_BIG_ENDIAN))
398  *Temp = ENDIAN_SWAP_16(*Temp);
399  #endif
400 
401  *Temperature = (float)((Temp[1] << 0x08) | Temp[0]) * 0.0625;
402 
403  return ONEWIRE_NO_ERROR;
404 }
