214

216  for(uint8_t i = 0x00; i < AES_DATASIZE; i++)

260 const AES_Error_t AES_CBC_Decrypt(uint8_t* EncryptedData, volatile uint8_t* DecryptedData, uint8_t* volatile Key, const uint8_t* InitVector, const uint16_t BlockCount)
261 {
262  uint8_t* KeyTemp;
263 
264  for(uint8_t Block = BlockCount; Block > 0; Block--)
265  {
266  // Reset the pointer for the key each block
267  KeyTemp = Key;
268  for(uint8_t i = 0x00; i < AES_KEYSIZE; i++)
269  {
270  AES.KEY = *(KeyTemp++);
271  }
272 
273  for(uint8_t i = 0x00; i < AES_DATASIZE; i++)
274  {
275  AES.STATE = *(EncryptedData++);
276  }
277 
278  // Set the module in decryption mode, enable XOR and start it
279  AES_SetDecryptMode();
280  AES_EnableCBC();
281 
282  while(AES_IsBusy());
283 
284  // Check for error
285  if(!(AES.STATUS & AES_ERROR_bm))
286  {
287  // Use the initialization vector for the first block
288  if(Block == BlockCount)
289  {
290  for(uint8_t i = 0; i < AES_DATASIZE; i++)
291  {
292  AES.STATE = *(InitVector++);
293  }
294  }
295  else
296  {
297  uint8_t* LastBlock = EncryptedData - (AES_DATASIZE << 1);
298  for(uint8_t i = 0x00; i < AES_DATASIZE; i++)
299  {
300  AES.STATE = *(LastBlock++);
301  }
302  }
303 
304  for(uint8_t i = 0x00; i < AES_DATASIZE; i++)
305  {
306  *(DecryptedData++) = AES.STATE;
307  }
308 
309  AES.CTRL = AES.CTRL & (~AES_XOR_bm);
310  }
311  else
312  {
313  return AES_ERROR;
314  }
315  }
316 
317  // Disable auto mode and XOR feature
318  AES_DisableCBC();
319 
320  return AES_NO_ERROR;
321 }
